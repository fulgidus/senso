# Plan 33-02: YAML Loader + Chest Registry + FastAPI Deps

## Goal

Implement the YAML manifest loader that parses + validates a `manifest.yaml` into a `DomainManifest`, implement `ChestRegistry` as a singleton that discovers and validates all declared Chests, and expose both via FastAPI dependency injection.

## must_haves

- `be/app/domain/loader.py` exports `load_manifest(path: Path) -> DomainManifest`
- `load_manifest` uses `yaml.safe_load` (never bare `yaml.load`)
- `load_manifest` validates the loaded dict through `DomainManifest.model_validate()`
- `load_manifest` raises `ManifestLoadError` (custom exception) with path + Pydantic error detail on validation failure
- `ChestRegistry.__init__` accepts a `DomainManifest` and indexes all chests by `id`
- `ChestRegistry.get(id)` raises `KeyError` with the unknown ID in the message
- `ChestRegistry.validate()` checks: no duplicate IDs, all `schema_ref` paths exist relative to manifest dir, all `hard_boundaries_ref` / `config_ref` paths exist; returns list of error strings (empty = valid)
- `be/app/domain/deps.py` exports `get_manifest(request)` and `get_chest_registry(request)` FastAPI deps
- `be/app/core/config.py` has `domain_manifest_path: str` field with default `"domains/finance/manifest.yaml"`
- `docker compose run --rm api python -c "from app.domain.loader import load_manifest; print('ok')"` exits 0

```python
"""
Domain Manifest loader.

Reads a YAML file, validates it against DomainManifest Pydantic models,
and returns a frozen DomainManifest instance.

Usage:
    from app.domain.loader import load_manifest
    manifest = load_manifest(Path("domains/finance/manifest.yaml"))
"""

from __future__ import annotations

import logging
from pathlib import Path
from typing import Any

import yaml
from pydantic import ValidationError

from app.domain.manifest import DomainManifest

logger = logging.getLogger(__name__)


class ManifestLoadError(Exception):
    """Raised when a Domain Manifest cannot be loaded or validated."""

    def __init__(self, path: Path, reason: str) -> None:
        self.path = path
        self.reason = reason
        super().__init__(f"Failed to load manifest at {path}: {reason}")


def load_manifest(path: Path | str) -> DomainManifest:
    """
    Load and validate a Domain Manifest from a YAML file.

    Args:
        path: Path to the manifest YAML file (absolute or relative to cwd).

    Returns:
        A frozen DomainManifest instance.

    Raises:
        ManifestLoadError: If the file cannot be read or fails Pydantic validation.
    """
    path = Path(path)
    if not path.exists():
        raise ManifestLoadError(path, f"file not found: {path}")

    try:
        raw: Any = yaml.safe_load(path.read_text(encoding="utf-8"))
    except yaml.YAMLError as exc:
        raise ManifestLoadError(path, f"YAML parse error: {exc}") from exc

    if not isinstance(raw, dict):
        raise ManifestLoadError(path, "manifest must be a YAML mapping at the top level")

    try:
        manifest = DomainManifest.model_validate(raw)
    except ValidationError as exc:
        # Surface the first error for quick diagnosis; full detail in exc.errors()
        first = exc.errors()[0]
        location = " → ".join(str(loc) for loc in first["loc"])
        raise ManifestLoadError(
            path,
            f"validation error at '{location}': {first['msg']}"
        ) from exc

    logger.info(
        "Manifest loaded: %s v%s — %d chest(s), %d extractor(s)",
        manifest.domain.name,
        manifest.domain.version,
        len(manifest.chests),
        len(manifest.extractors),
    )
    return manifest
```

```python
"""
Chest Registry — discovers, validates, and provides access to domain Chests.

Pattern mirrors ingestion/registry.py (ModuleRegistry singleton).
Built once at startup from the loaded DomainManifest; immutable after that.

Usage:
    from app.domain.registry import ChestRegistry
    registry = ChestRegistry(manifest, manifest_dir=Path("domains/finance"))
    chest = registry.get("user_profile")
    tool_chests = registry.tool_callable()
"""

from __future__ import annotations

import logging
from pathlib import Path

from app.domain.manifest import ChestDef, DomainManifest

logger = logging.getLogger(__name__)


class ChestRegistry:
    """
    Indexed view of all Chests declared in a Domain Manifest.

    Provides typed accessor methods used by platform subsystems:
      - Phase 34 (DataLayer) — storage config per chest
      - Phase 35 (ToolGen)   — tool_callable chests for LLM tool generation
      - Phase 35 (Search)    — searchable chests for BM25 indexing
      - Phase 36 (Filters)   — behavior + filter config from manifest
    """

    def __init__(self, manifest: DomainManifest, manifest_dir: Path | None = None) -> None:
        self._manifest = manifest
        self._manifest_dir = manifest_dir or Path(".")
        self._by_id: dict[str, ChestDef] = {}
        for chest in manifest.chests:
            if chest.id in self._by_id:
                raise ValueError(f"Duplicate chest ID in manifest: '{chest.id}'")
            self._by_id[chest.id] = chest
        logger.debug("ChestRegistry initialized with %d chests", len(self._by_id))

    # ── Accessor methods ──────────────────────────────────────────────────────

    def get(self, chest_id: str) -> ChestDef:
        """Return the ChestDef for the given ID.  Raises KeyError if not found."""
        try:
            return self._by_id[chest_id]
        except KeyError:
            available = ", ".join(sorted(self._by_id))
            raise KeyError(
                f"Chest '{chest_id}' not found in registry. "
                f"Available: [{available}]"
            ) from None

    def list(self, scope: str | None = None) -> list[ChestDef]:
        """Return all chests, optionally filtered by scope ('user'/'domain'/'platform')."""
        chests = list(self._by_id.values())
        if scope is not None:
            chests = [c for c in chests if c.scope == scope]
        return chests

    def tool_callable(self) -> list[ChestDef]:
        """Return chests with 'tool_callable' access (exposed to LLM tools)."""
        return [c for c in self._by_id.values() if "tool_callable" in c.access]

    def searchable(self) -> list[ChestDef]:
        """Return chests with 'searchable' access (indexed in BM25)."""
        return [c for c in self._by_id.values() if "searchable" in c.access]

    def ids(self) -> list[str]:
        """Return all registered chest IDs."""
        return list(self._by_id)

    # ── Validation ────────────────────────────────────────────────────────────

    def validate(self) -> list[str]:
        """
        Validate all chest and manifest references.

        Returns a list of error strings.  Empty list means valid.
        Warnings are logged but do NOT produce errors (e.g. module_ref importability).
        """
        errors: list[str] = []

        # 1. Validate chest schema_refs
        for chest in self._by_id.values():
            if chest.schema_ref:
                schema_path = self._manifest_dir / chest.schema_ref
                if not schema_path.exists():
                    errors.append(
                        f"Chest '{chest.id}': schema_ref not found: {schema_path}"
                    )

        # 2. Validate extractor module_refs (warn only — module may not be importable in all envs)
        for extractor in self._manifest.extractors:
            if extractor.type == "module" and extractor.module_ref:
                try:
                    import importlib
                    importlib.import_module(extractor.module_ref)
                except ImportError as exc:
                    logger.warning(
                        "Extractor '%s': module_ref '%s' not importable: %s",
                        extractor.id,
                        extractor.module_ref,
                        exc,
                    )

        # 3. Validate behavior refs
        behavior = self._manifest.behavior
        for attr, label in [
            ("ethos_ref", "behavior.ethos_ref"),
            ("boundaries_ref", "behavior.boundaries_ref"),
            ("allowlist_ref", "behavior.allowlist_ref"),
            ("hard_boundaries_ref", "behavior.hard_boundaries_ref"),
        ]:
            ref = getattr(behavior, attr)
            if ref:
                ref_path = self._manifest_dir / ref
                if not ref_path.exists():
                    errors.append(f"{label} not found: {ref_path}")

        # 4. Validate personas config_ref
        if self._manifest.personas and self._manifest.personas.config_ref:
            config_path = self._manifest_dir / self._manifest.personas.config_ref
            if not config_path.exists():
                errors.append(f"personas.config_ref not found: {config_path}")

        # 5. Validate enrichment schema_ref
        if self._manifest.enrichment.schema_ref:
            enrichment_path = self._manifest_dir / self._manifest.enrichment.schema_ref
            if not enrichment_path.exists():
                errors.append(
                    f"enrichment.schema_ref not found: {enrichment_path}"
                )

        return errors

    @property
    def manifest(self) -> DomainManifest:
        """The full DomainManifest this registry was built from."""
        return self._manifest

    @property
    def domain_id(self) -> str:
        return self._manifest.domain.id
```

```python
"""
FastAPI dependency functions for Domain Manifest and Chest Registry.

Usage:
    from app.domain.deps import get_manifest, get_chest_registry
    from app.domain.manifest import DomainManifest
    from app.domain.registry import ChestRegistry

    @router.get("/chests")
    def list_chests(registry: ChestRegistry = Depends(get_chest_registry)):
        return registry.ids()
"""

from __future__ import annotations

from fastapi import Request

from app.domain.manifest import DomainManifest
from app.domain.registry import ChestRegistry


def get_manifest(request: Request) -> DomainManifest:
    """FastAPI dep: returns the loaded DomainManifest from app.state."""
    return request.app.state.manifest  # type: ignore[no-any-return]


def get_chest_registry(request: Request) -> ChestRegistry:
    """FastAPI dep: returns the ChestRegistry from app.state."""
    return request.app.state.chest_registry  # type: ignore[no-any-return]
```
