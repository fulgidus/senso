```python
"""
NaCl / libsodium crypto helpers for Phase 13 crypto-identity foundation.

Key model:
  - Each user gets an X25519 key pair for Diffie-Hellman (future E2E message encryption).
  - Each user gets an Ed25519 key pair for signing (admin signed messages).
  - Public keys stored in users.public_key_b64 and users.signing_key_b64 (base64 TEXT).
  - Private keys are NEVER stored server-side after Phase 15 (client holds them).
    For Phase 13/14 demo: server-held private keys encrypted with user's existing
    AES-GCM wrapped key (users.encrypted_user_key).

Algorithms:
  - X25519 (curve25519): nacl.public.PrivateKey / PublicKey - key agreement
  - Ed25519: nacl.signing.SigningKey / VerifyKey - signatures
"""

from __future__ import annotations

import base64

import nacl.public
import nacl.signing


def generate_x25519_keypair() -> tuple[nacl.public.PrivateKey, nacl.public.PublicKey]:
    """Generate a fresh X25519 (curve25519) key pair for Diffie-Hellman.

    Returns:
        (private_key, public_key) - nacl.public objects.
        Call public_key_b64() / private_key_b64() to serialise for storage.
    """
    private_key = nacl.public.PrivateKey.generate()
    return private_key, private_key.public_key


def generate_ed25519_keypair() -> tuple[nacl.signing.SigningKey, nacl.signing.VerifyKey]:
    """Generate a fresh Ed25519 key pair for signing.

    Returns:
        (signing_key, verify_key) - nacl.signing objects.
        signing_key is 64 bytes (seed+pubkey concatenated in libsodium convention).
        verify_key is 32 bytes.
    """
    signing_key = nacl.signing.SigningKey.generate()
    return signing_key, signing_key.verify_key


# Alias used by admin user code for clarity
generate_admin_signing_keypair = generate_ed25519_keypair


def b64_encode(raw_bytes: bytes) -> str:
    """Base64-encode raw bytes to an ASCII string (no whitespace)."""
    return base64.b64encode(raw_bytes).decode("ascii")


def b64_decode(b64_str: str) -> bytes:
    """Decode a base64 ASCII string to raw bytes."""
    return base64.b64decode(b64_str)


def public_key_b64(pk: nacl.public.PublicKey) -> str:
    """Serialise an X25519 PublicKey to base64."""
    return b64_encode(bytes(pk))


def private_key_b64(sk: nacl.public.PrivateKey) -> str:
    """Serialise an X25519 PrivateKey to base64 (for wrapped-key storage)."""
    return b64_encode(bytes(sk))


def verify_key_b64(vk: nacl.signing.VerifyKey) -> str:
    """Serialise an Ed25519 VerifyKey to base64."""
    return b64_encode(bytes(vk))


def signing_key_b64(sk: nacl.signing.SigningKey) -> str:
    """Serialise an Ed25519 SigningKey to base64 (for wrapped-key storage)."""
    return b64_encode(bytes(sk))


def sign_message(message: bytes, sk_b64: str) -> str:
    """Sign a message with a base64-encoded Ed25519 signing key.

    Returns:
        Base64-encoded signature (64 bytes).
    """
    sk = nacl.signing.SigningKey(b64_decode(sk_b64))
    signed = sk.sign(message)
    return b64_encode(signed.signature)


def verify_signature(message: bytes, signature_b64: str, vk_b64: str) -> bool:
    """Verify an Ed25519 signature.

    Returns:
        True if valid, False if invalid.
    """
    try:
        vk = nacl.signing.VerifyKey(b64_decode(vk_b64))
        vk.verify(message, b64_decode(signature_b64))
        return True
    except Exception:
        return False
```

```python
"""
Username generation for Phase 13 crypto-identity foundation.

Format:
  - Regular users: `$adjective-noun-NNNN` e.g. `$witty-otter-42`
    - `$` prefix (distinguishes from !admin)
    - lowercase adjective + noun from curated word lists
    - 2-4 random decimal digits (not zero-padded)
  - Admin users: `!admin` (default), `!` prefix, renamable (any !word)

Collision resolution: re-roll digits up to 100 times, then extend word lists.
"""

from __future__ import annotations

import random
import string

from sqlalchemy.orm import Session

# ── Word lists ──────────────────────────────────────────────────────────────
ADJECTIVES: list[str] = [
    "witty", "brave", "calm", "deft", "eager",
    "fair", "glad", "hale", "idle", "keen",
    "lush", "mild", "neat", "odd", "pure",
    "quick", "rich", "sage", "tame", "vast",
    "warm", "zeal", "bold", "cozy", "deep",
    "epic", "firm", "gold", "hazy", "icy",
    "jade", "kind", "lean", "merry", "noble",
    "opal", "plum", "quiet", "rare", "swift",
]

NOUNS: list[str] = [
    "otter", "crane", "dove", "elk", "fox",
    "gecko", "hawk", "ibis", "jay", "kite",
    "lynx", "mink", "newt", "owl", "pika",
    "quail", "rook", "seal", "teal", "vole",
    "wolf", "wren", "yak", "zebu", "bee",
    "carp", "dart", "fern", "gull", "heron",
    "iris", "koi", "lark", "moth", "newb",
    "oryx", "pine", "reed", "ruse", "sage",
]

_MAX_RETRIES = 100


def generate_username(db: Session) -> str:
    """Generate a unique `$adjective-noun-NNNN` username.

    Queries users table to check for collisions. Retries up to 100 times.
    Raises RuntimeError if all retries exhausted (astronomically unlikely with 40×40×9999 space).

    Args:
        db: SQLAlchemy Session for uniqueness check.

    Returns:
        Unique username string beginning with `$`.
    """
    from app.db.models import User  # noqa: PLC0415 - avoid circular at module level

    for _ in range(_MAX_RETRIES):
        adj = random.choice(ADJECTIVES)
        noun = random.choice(NOUNS)
        digits = random.randint(10, 9999)
        candidate = f"${adj}-{noun}-{digits}"
        # Check uniqueness
        exists = db.query(User).filter(User.username == candidate).first()
        if exists is None:
            return candidate
    raise RuntimeError(
        "Username generator exhausted retries - username space may be full."
    )


def generate_admin_username() -> str:
    """Return the default admin username `!admin`.

    Admin usernames use `!` prefix. They are renamable after creation.
    """
    return "!admin"


def is_admin_username(username: str) -> bool:
    """Return True if username has the `!` prefix (admin identity)."""
    return username.startswith("!")


def is_user_username(username: str) -> bool:
    """Return True if username has the `$` prefix (regular user identity)."""
    return username.startswith("$")
```
