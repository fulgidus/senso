## Key Decisions

- Username: `$adjective-noun-NNNN` for regular users, `!admin` for admin users
- Crypto stack: X25519 (Diffie-Hellman) + Ed25519 (signing)
- Private keys are NOT stored in Phase 13 — only public keys stored in users table
- Phase 13 is server-key-generation only; client key management is Phase 15
- `email` column remains as primary login credential — it is not removed in this phase
- PII replacement scope in Phase 13: only `contributing_user_obfuscated` in admin merchant map (replace with `username`)
- UserDTO must expose `username`, `public_key_b64` (X25519), `signing_key_b64` (Ed25519 verify key)

```python
def signup(self, *, email: str, password: str) -> AuthResponseDTO:
    if repository.get_user_by_email(self.db, email) is not None:
        raise AuthError("email_in_use", "Email already registered", status_code=409)

    email_lower = email.lower()
    is_admin = email_lower in self.settings.starting_admins

    user = User(
        id=str(uuid4()),
        email=email_lower,
        password_hash=hash_password(password),
        is_admin=is_admin,
        role="admin" if is_admin else "user",
    )
    repository.create_user(self.db, user)

    # Phase 10: wrapped data key
    enc_key, salt_b64 = server_wrap_user_key(user.id)
    user.encrypted_user_key = enc_key
    user.pbkdf2_salt = salt_b64

    # Phase 13: crypto identity — username + NaCl key pairs
    user.username = generate_admin_username() if is_admin else generate_username(self.db)
    x_sk, x_pk = generate_x25519_keypair()
    ed_sk, ed_vk = generate_ed25519_keypair()
    user.public_key_b64 = x25519_pub_b64(x_pk)       # X25519 public key (stored)
    user.signing_key_b64 = ed25519_vk_b64(ed_vk)     # Ed25519 verify key (stored)
    # Private keys not stored server-side in Phase 13 (client key management is Phase 15)

    self.db.commit()
    return self._issue_auth_response(user)
```

```python
"""Integration tests for Phase 13 crypto-identity fields at signup."""
import re
import pytest
from fastapi.testclient import TestClient
from app.main import app


@pytest.fixture
def client():
    with TestClient(app) as c:
        yield c


def _signup(client, email: str, password: str = "password123"):
    return client.post("/auth/signup", json={"email": email, "password": password})


def test_signup_returns_username(client):
    resp = _signup(client, "newuser_phase13@example.com")
    assert resp.status_code == 200
    body = resp.json()
    username = body["user"]["username"]
    assert username is not None
    assert username.startswith("$"), f"Expected $ prefix, got: {username}"
    assert re.match(r"^\$[a-z]+-[a-z]+-\d+$", username), f"Bad format: {username}"


def test_signup_returns_public_keys(client):
    resp = _signup(client, "keysuser_phase13@example.com")
    assert resp.status_code == 200
    body = resp.json()
    user = body["user"]
    assert user["publicKeyB64"] is not None and len(user["publicKeyB64"]) > 0
    assert user["signingKeyB64"] is not None and len(user["signingKeyB64"]) > 0


def test_admin_signup_gets_admin_username(client, monkeypatch):
    """Admin email in starting_admins gets !admin username."""
    from app.core.config import get_settings
    settings = get_settings()
    admin_email = "admin_test_p13@senso.local"
    # Patch starting_admins to include this test email
    original = settings.starting_admins
    settings.__dict__["starting_admins"] = list(original) + [admin_email]
    try:
        resp = _signup(client, admin_email)
        assert resp.status_code == 200
        username = resp.json()["user"]["username"]
        assert username == "!admin", f"Expected !admin, got: {username}"
    finally:
        settings.__dict__["starting_admins"] = original


def test_login_returns_username_and_keys(client):
    """login() also returns username and key fields via _to_user_dto."""
    email = "logintest_phase13@example.com"
    _signup(client, email)
    resp = client.post("/auth/login", json={"email": email, "password": "password123"})
    assert resp.status_code == 200
    user = resp.json()["user"]
    assert user["username"] is not None
    assert user["publicKeyB64"] is not None
    assert user["signingKeyB64"] is not None


def test_me_endpoint_returns_username(client):
    """GET /auth/me returns username field for authenticated user."""
    email = "metest_phase13@example.com"
    signup_resp = _signup(client, email)
    token = signup_resp.json()["accessToken"]
    resp = client.get("/auth/me", headers={"Authorization": f"Bearer {token}"})
    assert resp.status_code == 200
    assert resp.json()["username"] is not None
```
