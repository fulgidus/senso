```python
"""
GenericCSV - fallback extractor for any CSV with date/amount/description columns.
Lowest priority (score floor 0.1). Handles Italian and English column name variants.
No institution-specific logic - best-effort column mapping only.
"""
import csv
from pathlib import Path
from decimal import Decimal
from datetime import datetime

FINGERPRINT: list[str] = ["date", "amount", "description"]  # very generic - low score by design
MODULE_VERSION: str = "1.0.0"

DATE_COLUMNS = ["date", "data", "data operazione", "transaction date", "completed date", "started date", "data valuta"]
AMOUNT_COLUMNS = ["amount", "importo", "entrate", "uscite", "accrediti", "addebiti", "amount (eur)"]
DESC_COLUMNS = ["description", "descrizione", "causale", "beneficiary / sender", "memo", "note", "dettaglio"]

def _find_col(headers: list[str], candidates: list[str]) -> int | None:
    headers_lower = [h.lower().strip() for h in headers]
    for candidate in candidates:
        for i, h in enumerate(headers_lower):
            if candidate in h:
                return i
    return None

def extract(file_path) -> dict:
    from app.schemas.ingestion import ExtractionResult, ExtractedDocument, Transaction
    path = Path(file_path)
    
    with open(str(path), newline="", encoding="utf-8-sig", errors="ignore") as f:
        reader = csv.reader(f)
        rows = list(reader)
    
    if not rows:
        return ExtractionResult(
            document=ExtractedDocument(document_type="bank_statement", module_name="GenericCSV", module_source="builtin", module_version=MODULE_VERSION),
            confidence=0.1, tier_used="module", warnings=["Empty CSV file"],
        )
    
    # Find header row
    header_row = rows[0]
    date_col = _find_col(header_row, DATE_COLUMNS)
    amount_col = _find_col(header_row, AMOUNT_COLUMNS)
    desc_col = _find_col(header_row, DESC_COLUMNS)
    
    if date_col is None or amount_col is None:
        return ExtractionResult(
            document=ExtractedDocument(document_type="bank_statement", module_name="GenericCSV", module_source="builtin", module_version=MODULE_VERSION),
            confidence=0.1, tier_used="module", warnings=["Could not detect required columns"],
        )
    
    transactions = []
    for row in rows[1:]:
        if not row or len(row) <= max(filter(None, [date_col, amount_col, desc_col or 0])):
            continue
        try:
            date_str = row[date_col].strip()
            amount_str = row[amount_col].strip().replace(",", ".").replace(" ", "").replace("€", "")
            if not date_str or not amount_str:
                continue
            
            # Try multiple date formats
            parsed_date = None
            for fmt in ["%Y-%m-%d", "%d/%m/%Y", "%m/%d/%Y", "%d-%m-%Y", "%Y/%m/%d"]:
                try:
                    parsed_date = datetime.strptime(date_str.split(" ")[0], fmt).date()
                    break
                except ValueError:
                    continue
            if parsed_date is None:
                continue
            
            amount = Decimal(amount_str).quantize(Decimal("0.0001"))
            desc = row[desc_col].strip() if desc_col is not None and desc_col < len(row) else ""
            
            transactions.append(Transaction(
                date=parsed_date,
                description=desc,
                amount=amount,
                currency="EUR",
            ))
        except Exception:
            continue
    
    return ExtractionResult(
        document=ExtractedDocument(
            document_type="bank_statement",
            module_name="GenericCSV",
            module_source="builtin",
            module_version=MODULE_VERSION,
            transactions=transactions,
        ),
        confidence=0.5 if transactions else 0.1,
        tier_used="module",
    )
```

```python
"""
EdisonEnergiaIT - Edison Energia Italy utility bill extractor.
Supported formats: PDF
Export path: Edison web portal → Bills → Download PDF
Sample keywords: [read from sample if available, else use known keywords]
"""
FINGERPRINT: list[str] = ["edison", "energia", "fornitura di energia"]  # verify with sample
MODULE_VERSION: str = "1.0.0"

def extract(file_path) -> dict:
    from app.schemas.ingestion import ExtractionResult, ExtractedDocument
    from pathlib import Path
    import re
    path = Path(file_path)
    
    raw_text = ""
    try:
        import pytesseract
        from PIL import Image
        from pdf2image import convert_from_path
        pages = convert_from_path(str(path), dpi=150)
        raw_text = "\n".join(pytesseract.image_to_string(p) for p in pages)
    except Exception:
        pass
    
    # Extract total_due from text
    total_due = None
    match = re.search(r"totale\s+(?:da\s+pagare|importo)[:\s]+€?\s*([\d.,]+)", raw_text, re.IGNORECASE)
    if match:
        from decimal import Decimal
        try:
            total_due = Decimal(match.group(1).replace(".", "").replace(",", "."))
        except Exception:
            pass
    
    # Detect service type
    service_type = "electricity"
    if "gas" in raw_text.lower() or "metano" in raw_text.lower():
        service_type = "gas"
    if "luce" in raw_text.lower() and "gas" in raw_text.lower():
        service_type = "electricity+gas"
    
    return ExtractionResult(
        document=ExtractedDocument(
            document_type="utility_bill",
            module_name="EdisonEnergiaIT",
            module_source="builtin",
            module_version=MODULE_VERSION,
            provider="Edison Energia",
            service_type=service_type,
            total_due=total_due,
            currency="EUR",
        ),
        confidence=0.7 if total_due else 0.4,
        raw_text=raw_text or None,
        tier_used="module",
    )
```
