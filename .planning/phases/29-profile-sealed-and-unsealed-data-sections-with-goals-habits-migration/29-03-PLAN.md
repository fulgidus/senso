```tsx
import { useCallback, useEffect, useState } from "react"
import { useTranslation } from "react-i18next"
import { Eye, Lock } from "lucide-react"
import { useAuthContext } from "@/features/auth/AuthContext"
import { createProfileApi } from "@/lib/profile-api"
import { sealForSelf, unsealFromSelf } from "@/features/messages/crypto"
import { useIsMobile } from "@/hooks/useIsMobile"

// ── TagInput (lifted verbatim from SettingsScreen) ────────────────────────────
function TagInput({
    items,
    onChange,
    placeholder,
    addButtonLabel,
}: {
    items: string[]
    onChange: (items: string[]) => void
    placeholder: string
    addButtonLabel: string
}) {
    const [input, setInput] = useState("")
    const isMobile = useIsMobile()

    const add = (val: string) => {
        const v = val.trim()
        if (v && !items.includes(v)) onChange([...items, v])
        setInput("")
    }

    return (
        <div className="flex flex-wrap gap-1">
            {items.map(item => (
                <span
                    key={item}
                    className="inline-flex items-center gap-1 rounded-full bg-primary/10 text-primary px-2.5 py-0.5 text-sm font-medium"
                >
                    {item}
                    <button
                        type="button"
                        onClick={() => onChange(items.filter(i => i !== item))}
                        className="ml-0.5 text-primary/70 hover:text-primary"
                        aria-label="Remove"
                    >
                        ×
                    </button>
                </span>
            ))}
            <input
                className="flex-1 min-w-[120px] rounded-lg border border-border bg-background px-2 py-1 text-sm focus:outline-none focus:ring-2 focus:ring-primary"
                value={input}
                placeholder={placeholder}
                onChange={e => setInput(e.target.value)}
                onKeyDown={e => {
                    if (e.key === "Enter" || e.key === ",") {
                        e.preventDefault()
                        add(input)
                    }
                }}
                onBlur={() => {
                    if (input.trim()) add(input)
                }}
            />
            {isMobile && (
                <button
                    type="button"
                    onClick={() => add(input)}
                    className="rounded-lg bg-muted px-3 py-1 text-sm text-muted-foreground hover:bg-muted/80"
                >
                    {addButtonLabel}
                </button>
            )}
        </div>
    )
}

// ── UnsealedPreferencesSection ────────────────────────────────────────────────
function UnsealedPreferencesSection({ token }: { token: string }) {
    const { t } = useTranslation()
    const { onUnauthorized } = useAuthContext()
    const api = createProfileApi(onUnauthorized)

    const [goals, setGoals] = useState<string[]>([])
    const [dos, setDos] = useState<string[]>([])
    const [donts, setDonts] = useState<string[]>([])
    const [saved, setSaved] = useState(false)

    useEffect(() => {
        api.getPreferences(token).then(data => {
            if (data) {
                setGoals(data.goals || [])
                setDos(data.dos || [])
                setDonts(data.donts || [])
            }
        })
    // eslint-disable-next-line react-hooks/exhaustive-deps
    }, [token])

    const save = useCallback(
        (g: string[], d: string[], dn: string[]) => {
            api.patchPreferences(token, g, d, dn).then(() => {
                setSaved(true)
                setTimeout(() => setSaved(false), 2000)
            })
        },
        // eslint-disable-next-line react-hooks/exhaustive-deps
        [token],
    )

    return (
        <section className="rounded-2xl border border-border bg-card p-6 space-y-4">
            <div className="flex items-center gap-2">
                <Eye className="h-4 w-4 text-muted-foreground" />
                <h2 className="text-base font-semibold">{t("profile.preferences.unsealedHeading")}</h2>
                {saved && (
                    <span className="text-xs text-primary ml-auto">{t("preferences.saved")}</span>
                )}
            </div>
            <p className="text-xs text-muted-foreground">{t("profile.preferences.unsealedHint")}</p>

            <div className="space-y-3">
                <div>
                    <label className="text-sm font-medium text-muted-foreground">
                        {t("preferences.goalsLabel")}
                    </label>
                    <TagInput
                        items={goals}
                        onChange={v => { setGoals(v); save(v, dos, donts) }}
                        placeholder={t("preferences.goalsPlaceholder")}
                        addButtonLabel={t("preferences.addButton")}
                    />
                </div>
                <div>
                    <label className="text-sm font-medium text-muted-foreground">
                        {t("preferences.dosLabel")}
                    </label>
                    <TagInput
                        items={dos}
                        onChange={v => { setDos(v); save(goals, v, donts) }}
                        placeholder={t("preferences.dosPlaceholder")}
                        addButtonLabel={t("preferences.addButton")}
                    />
                </div>
                <div>
                    <label className="text-sm font-medium text-muted-foreground">
                        {t("preferences.dontsLabel")}
                    </label>
                    <TagInput
                        items={donts}
                        onChange={v => { setDonts(v); save(goals, dos, v) }}
                        placeholder={t("preferences.dontsPlaceholder")}
                        addButtonLabel={t("preferences.addButton")}
                    />
                </div>
            </div>
        </section>
    )
}

// ── SealedNotesSection ────────────────────────────────────────────────────────
function SealedNotesSection({ token }: { token: string }) {
    const { t } = useTranslation()
    const { user, cryptoKeys, onUnauthorized } = useAuthContext()
    const api = createProfileApi(onUnauthorized)

    const [plaintext, setPlaintext] = useState("")
    const [saveState, setSaveState] = useState<"idle" | "saved" | "error">("idle")

    const isLocked = !user?.publicKeyB64 || !cryptoKeys

    // Decrypt on mount when keys are available
    useEffect(() => {
        if (isLocked) return
        api.getSealedNotes(token).then(data => {
            if (data?.ciphertext && user?.publicKeyB64 && cryptoKeys) {
                const decrypted = unsealFromSelf(
                    data.ciphertext,
                    user.publicKeyB64,
                    cryptoKeys.x25519PrivateKey,
                )
                if (decrypted !== null) setPlaintext(decrypted)
            }
        })
    // eslint-disable-next-line react-hooks/exhaustive-deps
    }, [token, isLocked])

    const handleBlur = useCallback(() => {
        if (isLocked || !user?.publicKeyB64) return
        const ciphertext = plaintext.length > 0
            ? sealForSelf(plaintext, user.publicKeyB64)
            : null
        api.patchSealedNotes(token, ciphertext)
            .then(() => {
                setSaveState("saved")
                setTimeout(() => setSaveState("idle"), 2000)
            })
            .catch(() => setSaveState("error"))
    // eslint-disable-next-line react-hooks/exhaustive-deps
    }, [plaintext, isLocked, token, user?.publicKeyB64])

    return (
        <section className="rounded-2xl border border-border bg-card p-6 space-y-4">
            <div className="flex items-center gap-2">
                <Lock className="h-4 w-4 text-muted-foreground" />
                <h2 className="text-base font-semibold">{t("profile.preferences.sealedHeading")}</h2>
            </div>
            <p className="text-xs text-muted-foreground">{t("profile.preferences.sealedHint")}</p>

            {isLocked ? (
                <div className="rounded-lg border border-border bg-muted p-4 flex items-center gap-3">
                    <Lock className="h-5 w-5 text-muted-foreground shrink-0" />
                    <div>
                        <p className="text-sm font-medium text-foreground">
                            {t("profile.preferences.lockedHeading")}
                        </p>
                        <p className="text-xs text-muted-foreground mt-0.5">
                            {t("profile.preferences.lockedBody")}
                        </p>
                    </div>
                </div>
            ) : (
                <div className="relative">
                    <textarea
                        value={plaintext}
                        onChange={e => setPlaintext(e.target.value)}
                        onBlur={handleBlur}
                        placeholder={t("profile.preferences.sealedPlaceholder")}
                        className="w-full rounded-lg border border-border bg-background px-3 py-2 text-sm text-foreground placeholder:text-muted-foreground resize-none min-h-[120px] focus:outline-none focus:ring-2 focus:ring-primary"
                    />
                    {saveState === "saved" && (
                        <span className="absolute bottom-2 right-2 text-xs text-primary">
                            {t("profile.preferences.savedIndicator")}
                        </span>
                    )}
                    {saveState === "error" && (
                        <span className="absolute bottom-2 right-2 text-xs text-destructive">
                            {t("profile.preferences.saveError")}
                        </span>
                    )}
                </div>
            )}
        </section>
    )
}

// ── PreferenzaTab ─────────────────────────────────────────────────────────────
export function PreferenzaTab({ token }: { token: string }) {
    return (
        <div className="space-y-6">
            <UnsealedPreferencesSection token={token} />
            <SealedNotesSection token={token} />
        </div>
    )
}
```
