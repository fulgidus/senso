```python
"""
PersonaEngine — domain-configurable persona loading and soul file resolution.

Replaces hardcoded PERSONAS_DIR paths in CoachingService.
Loaded at startup from manifest.personas.config_ref.

AI-04: persona engine loads soul files, style, voice config from Domain Manifest.
AI-06: coaching prompts assembled from domain-specific Jinja2 templates.
"""

from __future__ import annotations

import json
import logging
from pathlib import Path
from typing import Any

from jinja2 import Environment, FileSystemLoader, StrictUndefined

from app.domain.manifest import DomainManifest

logger = logging.getLogger(__name__)


class PersonaEngine:
    """
    Loads persona configuration and soul files from domain manifest paths.
    Provides Jinja2 environment for domain-specific prompt templates.
    """

    def __init__(self, manifest: DomainManifest, manifest_dir: Path) -> None:
        self._manifest_dir = manifest_dir
        self._personas: list[dict[str, Any]] = []
        self._soul_dir: Path | None = None

        if manifest.personas and manifest.personas.config_ref:
            config_path = manifest_dir / manifest.personas.config_ref
            if config_path.exists():
                config = json.loads(config_path.read_text(encoding="utf-8"))
                self._personas = config.get("personas", [])
                # Soul files live relative to the config file's directory
                self._soul_dir = config_path.parent / "soul"
                logger.info(
                    "PersonaEngine: loaded %d personas from %s",
                    len(self._personas), config_path,
                )
            else:
                logger.warning("PersonaEngine: personas.config_ref not found: %s", config_path)
        else:
            logger.warning("PersonaEngine: no personas.config_ref in manifest — persona loading disabled")

    def list_personas(self) -> list[dict[str, Any]]:
        """Return all persona dicts from the domain config."""
        return list(self._personas)

    def get_persona(self, persona_id: str) -> dict[str, Any] | None:
        """Return persona config dict by ID, or None if not found."""
        return next((p for p in self._personas if p.get("id") == persona_id), None)

    def load_soul(self, persona_id: str, locale: str = "it") -> str:
        """
        Load the soul file text for a persona + locale.

        Resolution order:
          1. soul/{persona_id}.{locale}.md
          2. soul/{persona_id}.md (default locale)
          3. Empty string with warning
        """
        if not self._soul_dir:
            logger.warning("PersonaEngine: no soul_dir — returning empty soul")
            return ""

        # Try locale-specific first, then default
        for filename in (f"{persona_id}.{locale}.md", f"{persona_id}.md"):
            soul_path = self._soul_dir / filename
            if soul_path.exists():
                return soul_path.read_text(encoding="utf-8")

        logger.warning(
            "PersonaEngine: soul file not found for persona '%s' locale '%s' in %s",
            persona_id, locale, self._soul_dir,
        )
        return ""

    def get_jinja_env(self, templates_dir: Path) -> Environment:
        """
        Return a Jinja2 Environment backed by the domain's templates directory.
        Falls back to the standard coaching prompts dir if domain-specific dir missing.
        """
        if templates_dir.exists():
            loader_path = str(templates_dir)
        else:
            # Fallback to built-in coaching prompts
            from pathlib import Path as _Path  # noqa: PLC0415
            loader_path = str(_Path(__file__).parent / "prompts")
            logger.warning(
                "PersonaEngine: templates dir '%s' not found — using built-in prompts",
                templates_dir,
            )
        return Environment(
            loader=FileSystemLoader(loader_path),
            undefined=StrictUndefined,
            autoescape=False,
        )

    def get_voice_id(
        self, persona_id: str | None, locale: str | None, gender: str | None
    ) -> str | None:
        """Delegate to existing personas/loader.py voice resolution."""
        from app.personas.loader import get_voice_id  # noqa: PLC0415
        return get_voice_id(persona_id, locale, gender)
```
