```python
@router.delete("/nuke")
def nuke(
    current_user: UserDTO = Depends(require_tester),
    db: Session = Depends(get_db),
) -> dict:
    """Delete ALL data for the current user (chat sessions+messages, extracted docs, uploads).
    Also resets the UserProfile confirmed flag and financial summary fields.

    Order respects FK constraints:
    1. ChatSession (CASCADE deletes ChatMessage, SessionParticipant via FK)
    2. ExtractedDocument (via upload_id FK)
    3. Upload
    4. UserProfile financial reset (set confirmed=False, null financial fields)
    """
    from app.db.models import ChatSession, UserProfile  # noqa: PLC0415

    # 1. Delete chat sessions (CASCADE deletes messages + participants)
    db.query(ChatSession).filter(ChatSession.owner_id == current_user.id).delete(
        synchronize_session=False
    )

    # 2. Delete extracted documents via upload FK
    upload_ids = [
        u.id
        for u in db.query(Upload.id).filter(Upload.user_id == current_user.id).all()
    ]
    if upload_ids:
        db.query(ExtractedDocument).filter(
            ExtractedDocument.upload_id.in_(upload_ids)
        ).delete(synchronize_session=False)

    # 3. Delete uploads
    db.query(Upload).filter(Upload.user_id == current_user.id).delete(
        synchronize_session=False
    )

    # 4. Reset UserProfile: clear financial summary + set confirmed=False
    profile = db.query(UserProfile).filter_by(user_id=current_user.id).first()
    if profile:
        profile.confirmed = False
        profile.income_summary = None
        profile.monthly_income = None
        profile.monthly_expenses = None
        profile.monthly_margin = None
        profile.balance = None
        profile.income_min = None
        profile.income_max = None
        profile.expense_min = None
        profile.expense_max = None
        profile.category_totals = {}
        profile.insight_cards = []
        profile.coaching_insights = []
        profile.transaction_count = 0
        profile.profile_generated_at = None

    db.commit()
    return {"nuked": True, "user_id": current_user.id}
```
