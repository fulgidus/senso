```tsx
import { useTranslation } from "react-i18next"
import { WifiOff } from "lucide-react"

export function OfflineBanner() {
  const isOnline = useOnlineStatus()
  const { t } = useTranslation()

  if (isOnline) return null

  return (
    <div
      role="alert"
      className="fixed top-14 left-0 right-0 z-[35] flex items-center justify-center gap-2 bg-destructive px-4 py-2 text-sm font-medium text-destructive-foreground"
    >
      <WifiOff className="h-4 w-4" aria-hidden="true" />
      {t("app.offlineBanner")}
    </div>
  )
}
```

**1. Create `senso/src/components/BalanceMask.tsx`:**

```tsx
import { useTranslation } from "react-i18next"

type BalanceMaskProps = {
  value: string | number
  masked: boolean
  className?: string
}

export function BalanceMask({ value, masked, className }: BalanceMaskProps) {
  const { t } = useTranslation()

  if (masked) {
    return (
      <span className={className} aria-label={t("accessibility.balanceHidden")}>
        ****
      </span>
    )
  }

  return <span className={className}>{value}</span>
}
```

**2. Update `senso/src/features/profile/ProfileScreen.tsx`:**

(Check if Eye/EyeOff are already imported from lucide-react)

b) Add state for balance masking at the top of the component:
```tsx
const [balanceMasked, setBalanceMasked] = useState(

```

c) Add toggle handler:
```tsx
const toggleBalanceMask = () => {
  setBalanceMasked((prev) => {
    const next = !prev
    localStorage.setItem("senso:balanceMask", String(next))
    return next
  })
}
```

d) Add an eye icon toggle button in the profile header area (near the profile title or summary section):

```tsx
<button

```

IMPORTANT: Not every number needs masking. Only mask values that represent actual euro amounts (income, expenses, balances, margins, transaction amounts). Do NOT mask:

```tsx
import { describe, it, expect, vi } from "vitest"
import { render, screen } from "@testing-library/react"
import { BalanceMask } from "./BalanceMask"

vi.mock("react-i18next", () => ({
  useTranslation: () => ({
    t: (key: string) => key,
    i18n: { language: "it" },
  }),
}))

describe("BalanceMask", () => {
  it("shows actual value when not masked", () => {
    render(<BalanceMask value="€1.234" masked={false} />)
    expect(screen.getByText("€1.234")).toBeDefined()
  })

  it("shows **** when masked", () => {
    render(<BalanceMask value="€1.234" masked={true} />)
    expect(screen.getByText("****")).toBeDefined()
    expect(screen.queryByText("€1.234")).toBeNull()
  })

  it("has aria-label when masked", () => {
    render(<BalanceMask value="€1.234" masked={true} />)
    const masked = screen.getByText("****")
    expect(masked.getAttribute("aria-label")).toBe("accessibility.balanceHidden")
  })
})
```
