# Plan 34-02: DomainManager + ChestAccessor + Startup Wiring

## Goal

Implement `DomainManager` (routing flag, table registry) and `ChestAccessor` (generic typed CRUD over dynamic Tables using SQLAlchemy Core API). Wire both into FastAPI lifespan after `DomainModelFactory` provisions tables.

## must_haves

- `DomainManager` tracks active domains and provides `get_table(domain_id, chest_id)`; raises `DomainInactiveError` if domain is disabled
- `ChestAccessor` uses SQLAlchemy Core (`table.select()`, `table.insert()`, etc.) — NOT ORM session.add()
- `ChestAccessor` returns plain `dict` (via `row._mapping`) — no Pydantic at this layer
- `ChestAccessor.list()` filters by `identity_id` for user-scoped chests; uses `list_domain()` for domain-scoped
- `ChestAccessor.list()` filters by `locale` when `StorageConfig.localized=True` and locale provided
- `ChestAccessor.create()` auto-populates: `id=uuid7`, `domain_id`, `created_at`, `updated_at`; `expires_at` set from `StorageConfig.ttl_days` if set
- `main.py` lifespan provisions tables and activates domain after manifest load
- `app.state.domain_manager` available after startup
- `docker compose run --rm api python -c "from app.data.manager import DomainManager; print('ok')"` exits 0

```python
"""
DomainManager — routing flag and table registry for provisioned domains.

Domains are provisioned at startup (tables created).
Enabled/disabled at runtime via activate()/deactivate() — zero DDL.
"""

from __future__ import annotations

import logging
from sqlalchemy import Table

logger = logging.getLogger(__name__)


class DomainInactiveError(RuntimeError):
    """Raised when a request targets a domain that is not active."""


class DomainManager:
    """
    Tracks which domains are active and provides access to their Tables.

    Tables are registered once at startup after DomainModelFactory.provision().
    Active flag is flipped at runtime (enable/disable) without any DDL.
    """

    def __init__(self) -> None:
        self._active: set[str] = set()
        # domain_id → { chest_id → Table }
        self._tables: dict[str, dict[str, Table]] = {}

    # ── Registration (startup only) ───────────────────────────────────────────

    def register(self, domain_id: str, tables: dict[str, Table]) -> None:
        """Register all chest Tables for a domain. Called once at startup."""
        self._tables[domain_id] = dict(tables)
        logger.debug("DomainManager: registered domain '%s' (%d tables)", domain_id, len(tables))

    def activate(self, domain_id: str) -> None:
        """Mark domain as active — start accepting requests."""
        if domain_id not in self._tables:
            raise KeyError(f"Domain '{domain_id}' not registered — call register() first")
        self._active.add(domain_id)
        logger.info("DomainManager: activated domain '%s'", domain_id)

    def deactivate(self, domain_id: str) -> None:
        """Mark domain as inactive — stop routing (no DDL, tables remain)."""
        self._active.discard(domain_id)
        logger.info("DomainManager: deactivated domain '%s'", domain_id)

    # ── Runtime access ────────────────────────────────────────────────────────

    def is_active(self, domain_id: str) -> bool:
        return domain_id in self._active

    def get_table(self, domain_id: str, chest_id: str) -> Table:
        """
        Return the SQLAlchemy Table for (domain_id, chest_id).

        Raises:
            DomainInactiveError: if domain is not active.
            KeyError: if domain or chest_id is not registered.
        """
        if not self.is_active(domain_id):
            raise DomainInactiveError(
                f"Domain '{domain_id}' is not active. "
                "Call activate() or check manifest loading."
            )
        try:
            return self._tables[domain_id][chest_id]
        except KeyError:
            available = list(self._tables.get(domain_id, {}).keys())
            raise KeyError(
                f"Chest '{chest_id}' not found in domain '{domain_id}'. "
                f"Available: {available}"
            ) from None

    def active_domains(self) -> list[str]:
        return list(self._active)

    def registered_domains(self) -> list[str]:
        return list(self._tables.keys())
```

```python
"""
ChestAccessor — generic CRUD over a dynamically generated chest Table.

Uses SQLAlchemy Core API (table.select(), table.insert(), etc.).
Returns plain dict rows — no ORM models, no Pydantic at this layer.
Domain services shape the dicts into domain-specific types above this layer.
"""

from __future__ import annotations

import logging
from datetime import UTC, datetime, timedelta
from typing import Any

import uuid_utils as uuid
from sqlalchemy import Table, delete, select, update
from sqlalchemy.orm import Session

from app.domain.manifest import ChestDef

logger = logging.getLogger(__name__)


def _now() -> datetime:
    return datetime.now(UTC)


def _uuid7() -> str:
    return str(uuid.uuid7())


class ChestAccessor:
    """
    Generic typed CRUD for a single chest Table.

    All reads/writes go through SQLAlchemy Core — no ORM session.add().
    Returns plain dict objects (row._mapping converted to dict).
    """

    def __init__(self, table: Table, chest_def: ChestDef) -> None:
        self._table = table
        self._def = chest_def

    # ── Read ──────────────────────────────────────────────────────────────────

    def get(self, db: Session, id: str) -> dict[str, Any] | None:
        """Fetch one row by primary key."""
        row = db.execute(
            select(self._table).where(self._table.c.id == id)
        ).first()
        return dict(row._mapping) if row else None

    def list(
        self,
        db: Session,
        identity_id: str,
        locale: str | None = None,
    ) -> list[dict[str, Any]]:
        """
        List rows for a user-scoped chest.
        Filters by identity_id; optionally filters by locale if localized.
        """
        stmt = select(self._table).where(
            self._table.c.identity_id == identity_id
        )
        if self._def.storage.localized and locale:
            stmt = stmt.where(self._table.c.locale == locale)
        # Exclude expired rows
        stmt = stmt.where(
            (self._table.c.expires_at == None) |  # noqa: E711
            (self._table.c.expires_at > _now())
        )
        rows = db.execute(stmt).all()
        return [dict(r._mapping) for r in rows]

    def list_domain(
        self,
        db: Session,
        domain_id: str,
        locale: str | None = None,
    ) -> list[dict[str, Any]]:
        """List rows for a domain-scoped chest (no identity_id filter)."""
        stmt = select(self._table).where(
            self._table.c.domain_id == domain_id
        )
        if self._def.storage.localized and locale:
            stmt = stmt.where(self._table.c.locale == locale)
        stmt = stmt.where(
            (self._table.c.expires_at == None) |  # noqa: E711
            (self._table.c.expires_at > _now())
        )
        rows = db.execute(stmt).all()
        return [dict(r._mapping) for r in rows]

    def count(self, db: Session, identity_id: str) -> int:
        from sqlalchemy import func
        result = db.execute(
            select(func.count()).select_from(self._table).where(
                self._table.c.identity_id == identity_id
            )
        ).scalar()
        return result or 0

    # ── Write ─────────────────────────────────────────────────────────────────

    def create(
        self,
        db: Session,
        data: dict[str, Any],
        identity_id: str | None = None,
        domain_id: str = "",
        locale: str | None = None,
    ) -> str:
        """
        Insert a new row. Returns the generated id.

        identity_id: required for user-scoped chests; None for domain-scoped.
        domain_id: the active domain (e.g. 'personal_finance').
        """
        now = _now()
        expires_at = None
        if self._def.storage.ttl_days:
            expires_at = now + timedelta(days=self._def.storage.ttl_days)

        row_id = _uuid7()
        row = {
            "id": row_id,
            "identity_id": identity_id,
            "domain_id": domain_id,
            "locale": locale,
            "created_at": now,
            "updated_at": now,
            "expires_at": expires_at,
            **data,
        }
        db.execute(self._table.insert().values(**row))
        return row_id

    def update(self, db: Session, id: str, data: dict[str, Any]) -> bool:
        """Update domain fields for an existing row. Returns True if row existed."""
        result = db.execute(
            update(self._table)
            .where(self._table.c.id == id)
            .values(updated_at=_now(), **data)
        )
        return result.rowcount > 0

    def delete(self, db: Session, id: str) -> bool:
        """Hard-delete a row. Returns True if row existed."""
        result = db.execute(
            delete(self._table).where(self._table.c.id == id)
        )
        return result.rowcount > 0

    def upsert_by_identity(
        self,
        db: Session,
        identity_id: str,
        data: dict[str, Any],
        domain_id: str = "",
        locale: str | None = None,
    ) -> str:
        """
        Upsert: update existing row for identity_id if exists, else create.
        Useful for single-row-per-user chests (user_profile, user_prefs).
        Returns the row id.
        """
        stmt = select(self._table.c.id).where(
            self._table.c.identity_id == identity_id
        )
        if locale:
            stmt = stmt.where(self._table.c.locale == locale)
        existing = db.execute(stmt).first()
        if existing:
            self.update(db, existing.id, data)
            return existing.id
        return self.create(db, data, identity_id=identity_id, domain_id=domain_id, locale=locale)

    @property
    def table(self) -> Table:
        return self._table
```

```python
"""FastAPI dependency functions for DomainManager and ChestAccessor."""

from __future__ import annotations

from fastapi import Request

from app.data.accessor import ChestAccessor
from app.data.manager import DomainManager
from app.domain.registry import ChestRegistry


def get_domain_manager(request: Request) -> DomainManager:
    return request.app.state.domain_manager  # type: ignore[no-any-return]


def get_chest_accessor(chest_id: str, request: Request) -> ChestAccessor:
    """
    FastAPI dep: returns ChestAccessor for the given chest_id in the active domain.

    Usage:
        from functools import partial
        from fastapi import Depends
        get_transactions = partial(get_chest_accessor, "transactions")
        @router.get("/transactions")
        def list_transactions(accessor = Depends(get_transactions)): ...
    """
    manager: DomainManager = request.app.state.domain_manager
    registry: ChestRegistry = request.app.state.chest_registry
    manifest = request.app.state.manifest
    table = manager.get_table(manifest.domain.id, chest_id)
    chest_def = registry.get(chest_id)
    return ChestAccessor(table, chest_def)
```

```python
        raise RuntimeError(str(exc)) from exc

    from app.db.session import engine as _engine  # noqa: PLC0415
    _factory.provision(_engine, _domain_tables)

    app.state.domain_manager = _domain_manager

        _manifest.domain.id,
        len(_domain_tables),
```
