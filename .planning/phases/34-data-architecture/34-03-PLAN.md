# Plan 34-03: ChestSearchEngine (BM25)

## Goal

Implement `ChestSearchEngine` — a locale-partitioned BM25 index over all `searchable` Chests. Mirrors the existing `ContentIndex` pattern but is generalized to any chest type. Wire into FastAPI lifespan after table provisioning.

## must_haves

- `ChestSearchEngine` in `be/app/data/search.py` builds one BM25 index per `(chest_id, locale)` partition
- `search(chest_id, query, locale, top_k)` returns `list[dict]` with a `_score` field added
- `build_chest(chest_id, items, locale, text_fields)` indexes arbitrary dicts — no coupling to specific domain schema
- `text_fields` auto-derived from chest schema: all string-type properties excluding platform columns (`id`, `identity_id`, `domain_id`, `locale`, `created_at`, `updated_at`, `expires_at`)
- Engine rebuilds all searchable chests at startup via `rebuild_all(registry, domain_manager, manifest, db)`
- `app.state.chest_search` available after startup
- Empty chest (zero rows) produces no index (no error) — search returns `[]`
- `docker compose run --rm api python -c "from app.data.search import ChestSearchEngine; print('ok')"` exits 0

```python
"""
ChestSearchEngine — locale-partitioned BM25 search over domain Chests.

Generalizes ContentIndex from be/app/content/search.py to cover any
searchable Chest type.  Pattern is identical: one BM25Okapi index per
(chest_id, locale) partition, rebuilt at startup from DB rows.

Usage:
    engine = ChestSearchEngine()
    engine.build_chest("content", items, locale="it", text_fields=["title", "summary", "body"])
    results = engine.search("content", "risparmio pensione", locale="it", top_k=5)
"""

from __future__ import annotations

import logging
import re
from typing import Any

from rank_bm25 import BM25Okapi  # type: ignore[import-untyped]

logger = logging.getLogger(__name__)

# ── Tokenizer (same as content/search.py) ────────────────────────────────────

_STOP_WORDS = frozenset(
    "il lo la le gli i un una e o ma se non è con per da in su tra fra "
    "the a an and or but if not is with for from in on at".split()
)


def _tokenize(text: str) -> list[str]:
    """Lowercase, strip punctuation, remove stop words."""
    tokens = re.findall(r"[a-zA-ZÀ-ÿ0-9]+", text.lower())
    return [t for t in tokens if t not in _STOP_WORDS and len(t) > 1]


# ── Platform columns to exclude from text indexing ───────────────────────────

_PLATFORM_COLUMNS = frozenset(
    {"id", "identity_id", "domain_id", "locale", "created_at", "updated_at", "expires_at"}
)


def _item_to_text(item: dict[str, Any], text_fields: list[str]) -> str:
    """Concatenate text fields from a chest row into a single searchable string."""
    parts: list[str] = []
    for field in text_fields:
        val = item.get(field)
        if val and isinstance(val, str):
            parts.append(val)
    return " ".join(parts)


def _derive_text_fields(schema: dict[str, Any]) -> list[str]:
    """
    Derive text fields from a JSON Schema.
    Returns all string-type top-level properties not in _PLATFORM_COLUMNS.
    """
    fields: list[str] = []
    for prop_name, prop_schema in schema.get("properties", {}).items():
        if prop_name in _PLATFORM_COLUMNS:
            continue
        if prop_schema.get("type") == "string" and prop_schema.get("format") is None:
            fields.append(prop_name)
    return fields


# ── Engine ────────────────────────────────────────────────────────────────────


class ChestSearchEngine:
    """
    BM25 search engine over multiple chest types.

    Index structure:
        { (chest_id, locale): (items: list[dict], bm25: BM25Okapi) }

    Rebuilt at startup; call rebuild_all() to refresh after data changes.
    """

    def __init__(self) -> None:
        self._indexes: dict[tuple[str, str], tuple[list[dict[str, Any]], BM25Okapi]] = {}

    def build_chest(
        self,
        chest_id: str,
        items: list[dict[str, Any]],
        locale: str,
        text_fields: list[str],
    ) -> None:
        """
        Build BM25 index for one (chest_id, locale) partition.

        items: rows from ChestAccessor (plain dicts)
        text_fields: schema string fields to concatenate for BM25 corpus
        """
        if not items:
            logger.debug("ChestSearchEngine: no items for (%s, %s) — skipping", chest_id, locale)
            return

        corpus = [_tokenize(_item_to_text(item, text_fields)) for item in items]
        bm25 = BM25Okapi(corpus)
        self._indexes[(chest_id, locale)] = (items, bm25)
        logger.info(
            "ChestSearchEngine: indexed %d items for (%s, %s)",
            len(items),
            chest_id,
            locale,
        )

    def search(
        self,
        chest_id: str,
        query: str,
        locale: str,
        top_k: int = 5,
    ) -> list[dict[str, Any]]:
        """
        BM25 search over a chest partition.
        Returns top_k items sorted by score, with '_score' field added.
        Returns [] if chest has no index for this locale.
        """
        key = (chest_id, locale)
        if key not in self._indexes:
            logger.debug("ChestSearchEngine: no index for %s — returning empty", key)
            return []

        items, bm25 = self._indexes[key]
        tokens = _tokenize(query)
        if not tokens:
            return []

        scores = bm25.get_scores(tokens)
        ranked = sorted(
            zip(scores, items),
            key=lambda x: x[0],
            reverse=True,
        )[:top_k]

        results = []
        for score, item in ranked:
            if score > 0:
                results.append({**item, "_score": round(float(score), 4)})
        return results

    def rebuild_all(
        self,
        manifest,       # DomainManifest
        manifest_dir,   # Path
        manager,        # DomainManager
        db,             # Session
        locales: list[str] | None = None,
    ) -> None:
        """
        Rebuild BM25 indexes for all searchable chests in the manifest.

        locales: if None, derives locales from existing rows.
                 If provided, builds indexes for these locales.
        """
        import json
        from pathlib import Path
        from app.data.accessor import ChestAccessor
        from app.domain.registry import ChestRegistry

        _locales = locales or [manifest.domain.locale_default]

        for chest_def in manifest.chests:
            if "searchable" not in chest_def.access:
                continue

            # Derive text fields from schema
            text_fields: list[str] = []
            if chest_def.schema_ref:
                schema_path = Path(manifest_dir) / chest_def.schema_ref
                if schema_path.exists():
                    schema = json.loads(schema_path.read_text(encoding="utf-8"))
                    text_fields = _derive_text_fields(schema)
            if not text_fields:
                logger.warning(
                    "ChestSearchEngine: chest '%s' has no text fields — skipping BM25",
                    chest_def.id,
                )
                continue

            try:
                table = manager.get_table(manifest.domain.id, chest_def.id)
                accessor = ChestAccessor(table, chest_def)
            except Exception as exc:
                logger.warning("ChestSearchEngine: cannot access chest '%s': %s", chest_def.id, exc)
                continue

            for locale in _locales:
                try:
                    if chest_def.scope == "domain":
                        items = accessor.list_domain(db, manifest.domain.id, locale=locale)
                    else:
                        # For user-scoped searchable chests: index all rows across users
                        # (these are typically content/knowledge chests in practice)
                        from sqlalchemy import select
                        rows = db.execute(
                            select(table).where(table.c.locale == locale)
                            if chest_def.storage.localized
                            else select(table)
                        ).all()
                        items = [dict(r._mapping) for r in rows]

                    self.build_chest(chest_def.id, items, locale, text_fields)
                except Exception as exc:
                    logger.warning(
                        "ChestSearchEngine: failed to build index for (%s, %s): %s",
                        chest_def.id, locale, exc,
                    )

    def indexed_partitions(self) -> list[tuple[str, str]]:
        """Return list of (chest_id, locale) partitions currently indexed."""
        return list(self._indexes.keys())
```

```python
    from app.data.search import ChestSearchEngine  # noqa: PLC0415

    _search_engine = ChestSearchEngine()

                manifest_dir=_manifest_path.parent,
```
