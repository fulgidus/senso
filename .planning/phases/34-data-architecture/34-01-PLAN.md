```python
"""
DomainModelFactory — generates typed SQLAlchemy Tables from Chest JSON Schemas.

Design rules:
  - Flat schemas only. Arrays and objects raise SchemaFlattenError.
  - Unknown types raise SchemaTypeError. NEVER fall back to JSONB.
  - Standard platform columns are always prepended (identity_id, domain_id, etc.).
  - PII encryption (StorageConfig.encrypted=True) wraps domain columns in
    StringEncryptedType; standard columns are always plaintext.
"""

from __future__ import annotations

import json
import logging
from pathlib import Path
from typing import Any, Callable

from sqlalchemy import (
    Boolean,
    Column,
    Date,
    DateTime,
    Integer,
    MetaData,
    Numeric,
    String,
    Table,
    Text,
    Engine,
)
from sqlalchemy_utils import StringEncryptedType
from sqlalchemy_utils.types.encrypted.encrypted_type import AesGcmEngine

from app.domain.manifest import ChestDef, DomainManifest, StorageConfig

logger = logging.getLogger(__name__)

import uuid_utils as uuid


def _uuid7() -> str:
    return str(uuid.uuid7())


# ── Type mapping ──────────────────────────────────────────────────────────────

# (json_type, format_or_None) → SQLAlchemy type constructor (not instance)
_TYPE_MAP: dict[tuple[str, str | None], Any] = {
    ("string",  None):          Text,
    ("string",  "date"):        Date,
    ("string",  "date-time"):   lambda: DateTime(timezone=True),
    ("string",  "uuid"):        lambda: String(36),
    ("string",  "email"):       lambda: String(255),
    ("number",  None):          Numeric,
    ("integer", None):          Integer,
    ("boolean", None):          Boolean,
}


def _resolve_type(prop_name: str, prop_schema: dict[str, Any]) -> Any:
    """
    Map a JSON Schema property to a SQLAlchemy column type.

    Raises:
        SchemaFlattenError: if property type is 'object' or 'array'.
        SchemaTypeError: if type+format combination is unsupported.
    """
    json_type = prop_schema.get("type")
    fmt = prop_schema.get("format")

    if json_type in ("object", "array"):
        raise SchemaFlattenError(
            f"Property '{prop_name}' has type '{json_type}'. "
            "Chest schemas must be flat — split into a separate chest or junction table."
        )

    key = (json_type, fmt)
    if key not in _TYPE_MAP:
        # Try without format before failing
        key_no_fmt = (json_type, None)
        if key_no_fmt not in _TYPE_MAP:
            raise SchemaTypeError(
                f"Property '{prop_name}': unsupported type='{json_type}' format='{fmt}'. "
                f"Supported: {sorted(set(k[0] for k in _TYPE_MAP))}. "
                "Add a mapping to _TYPE_MAP or simplify the schema."
            )
        key = key_no_fmt

    type_ctor = _TYPE_MAP[key]
    return type_ctor() if callable(type_ctor) and not isinstance(type_ctor, type) else type_ctor


# ── Exceptions ────────────────────────────────────────────────────────────────


class SchemaFlattenError(ValueError):
    """Raised when a Chest schema contains nested objects or arrays."""


class SchemaTypeError(ValueError):
    """Raised when a Chest schema property has an unsupported JSON Schema type."""


# ── Standard platform columns (always present on every chest table) ───────────

def _standard_columns() -> list[Column]:
    """Return the platform columns prepended to every chest table."""
    return [
        Column("id", String(36), primary_key=True, default=_uuid7),
        Column(
            "identity_id",
            String(36),
            nullable=True,   # NULL for scope='domain' chests
            index=True,
        ),
        Column("domain_id", String(64), nullable=False, index=True),
        Column("locale", String(10), nullable=True),
        Column(
            "created_at",
            DateTime(timezone=True),
            nullable=False,
            default=lambda: __import__("datetime").datetime.now(
                __import__("datetime").timezone.utc
            ),
        ),
        Column(
            "updated_at",
            DateTime(timezone=True),
            nullable=False,
            default=lambda: __import__("datetime").datetime.now(
                __import__("datetime").timezone.utc
            ),
            onupdate=lambda: __import__("datetime").datetime.now(
                __import__("datetime").timezone.utc
            ),
        ),
        Column("expires_at", DateTime(timezone=True), nullable=True),
    ]


# ── Factory ───────────────────────────────────────────────────────────────────


class DomainModelFactory:
    """
    Generates typed SQLAlchemy Table objects from Chest JSON Schemas.

    Usage:
        factory = DomainModelFactory(metadata)
        tables = factory.build_tables(manifest, manifest_dir, key_fn)
        factory.provision(engine, tables)
    """

    def __init__(self, metadata: MetaData) -> None:
        self._metadata = metadata

    def _load_schema(self, chest: ChestDef, manifest_dir: Path) -> dict[str, Any]:
        """Load and return the JSON Schema for a chest."""
        if not chest.schema_ref:
            logger.warning(
                "Chest '%s' has no schema_ref — only standard platform columns will be created.",
                chest.id,
            )
            return {"properties": {}, "required": []}
        schema_path = manifest_dir / chest.schema_ref
        if not schema_path.exists():
            raise FileNotFoundError(
                f"Chest '{chest.id}': schema_ref not found at {schema_path}"
            )
        return json.loads(schema_path.read_text(encoding="utf-8"))

    def _build_domain_columns(
        self,
        chest: ChestDef,
        schema: dict[str, Any],
        key_fn: Callable[[], str],
    ) -> list[Column]:
        """Build domain-schema columns from JSON Schema properties."""
        properties: dict[str, Any] = schema.get("properties", {})
        required: set[str] = set(schema.get("required", []))
        columns: list[Column] = []

        for prop_name, prop_schema in properties.items():
            sa_type = _resolve_type(prop_name, prop_schema)
            nullable = prop_name not in required

            if chest.storage.encrypted:
                # Wrap in AES-GCM encryption — stores as TEXT ciphertext
                col_type = StringEncryptedType(Text, key_fn, AesGcmEngine)
            else:
                col_type = sa_type

            # Add B-tree index for non-encrypted non-text columns (hot query paths)
            indexed = (
                not chest.storage.encrypted
                and not isinstance(sa_type, type(Text))
                and sa_type not in (Text,)
            )

            columns.append(
                Column(prop_name, col_type, nullable=nullable, index=indexed)
            )

        return columns

    def build_table(
        self,
        chest: ChestDef,
        domain_id: str,
        manifest_dir: Path,
        key_fn: Callable[[], str],
    ) -> Table:
        """Build a single typed Table for one ChestDef."""
        table_name = f"{domain_id}__{chest.id}"
        schema = self._load_schema(chest, manifest_dir)
        domain_cols = self._build_domain_columns(chest, schema, key_fn)
        all_cols = _standard_columns() + domain_cols

        # SQLAlchemy Table constructor: positional args are name, metadata, *columns
        return Table(table_name, self._metadata, *all_cols, extend_existing=True)

    def build_tables(
        self,
        manifest: DomainManifest,
        manifest_dir: Path,
        key_fn: Callable[[], str],
    ) -> dict[str, Table]:
        """
        Build one Table per ChestDef in the manifest.

        Returns:
            {chest_id: Table}

        Raises:
            SchemaFlattenError: if any schema has nested object/array properties.
            SchemaTypeError: if any schema has an unsupported type.
            FileNotFoundError: if a schema_ref file is missing.
        """
        tables: dict[str, Table] = {}
        for chest in manifest.chests:
            table = self.build_table(chest, manifest.domain.id, manifest_dir, key_fn)
            tables[chest.id] = table
            logger.debug(
                "Built table '%s' with %d domain columns",
                table.name,
                len(table.columns) - len(_standard_columns()),
            )
        return tables

    def provision(self, engine: Engine, tables: dict[str, Table]) -> None:
        """
        CREATE TABLE IF NOT EXISTS for all provided tables.
        Idempotent — safe to call on every startup.
        """
        self._metadata.create_all(engine, tables=list(tables.values()))
        for table_name in tables:
            logger.info("Provisioned table: %s", tables[table_name].name)
```
