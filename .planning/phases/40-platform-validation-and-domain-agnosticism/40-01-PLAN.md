"""
Platform agnosticism audit — CORE-06.

Verifies platform core has zero domain-specific code.
All domain logic must live in manifest + plugins (be/domains/).

Phase 40 success criteria:
  1. Platform core has zero refs to domain concepts: email, password, persona_id,
     voice_gender, finance, transaction, income — any domain concept
  2. platform_identities is the only identity table in platform migrations
"""

import re
from pathlib import Path

import pytest

# Platform core packages to audit (must be domain-free)
PLATFORM_PACKAGES = [
    "be/app/domain",
    "be/app/data",
    "be/app/platform",
    "be/app/tools",
]

# Domain-specific terms forbidden in platform code
FORBIDDEN_TERMS = [
    r"\bfinance\b",
    r"\bpersonal_finance\b",
    r"\btransaction\b",
    r"\bincome\b",
    r"\bpayslip\b",
    r"\bbank_statement\b",
    r"\bmentore.saggio\b",
    r"\bcheerleader\b",
    r"\bhartman\b",
    r"voice_gender",
    r"persona_id",
    r"first_name",
    r"last_name",
]

# Allowed exceptions (platform code that must mention these for valid reasons)
ALLOWED_EXCEPTIONS: dict[str, list[str]] = {
    # platform_identities is allowed — it's the platform primitive
    "platform_identities": ["be/app/db/models.py", "be/app/db/session.py"],
}


def _collect_python_files(package_dir: str) -> list[Path]:
    root = Path(package_dir)
    if not root.exists():
        return []
    return [p for p in root.rglob("*.py") if "__pycache__" not in str(p)]


@pytest.mark.parametrize("package", PLATFORM_PACKAGES)
def test_no_domain_terms_in_platform_package(package: str) -> None:
    """Platform package must contain zero domain-specific terms."""
    files = _collect_python_files(package)
    if not files:
        pytest.skip(f"Package not yet created: {package}")

    violations: list[str] = []
    for py_file in files:
        content = py_file.read_text(encoding="utf-8")
        for term_pattern in FORBIDDEN_TERMS:
            for match in re.finditer(term_pattern, content, re.IGNORECASE):
                line_num = content[:match.start()].count("\n") + 1
                violations.append(
                    f"{py_file}:{line_num}: found forbidden term '{match.group()}'"
                )

    assert not violations, (
        f"Platform package '{package}' contains domain-specific code:\n"
        + "\n".join(violations[:20])
        + (f"\n... and {len(violations)-20} more" if len(violations) > 20 else "")
    )


def test_platform_identities_is_only_identity_table() -> None:
    """platform_identities must be the only platform-level identity table."""
    models_file = Path("be/app/db/models.py")
    if not models_file.exists():
        pytest.skip("models.py not yet migrated")

    content = models_file.read_text(encoding="utf-8")

    # PlatformIdentity must exist
    assert "class PlatformIdentity" in content, "PlatformIdentity not found in models.py"

    # platform_identities must be the tablename
    assert "__tablename__ = \"platform_identities\"" in content

    # platform_identities table must have exactly 3 columns
    import ast
    tree = ast.parse(content)
    for node in ast.walk(tree):
        if isinstance(node, ast.ClassDef) and node.name == "PlatformIdentity":
            column_assigns = [
                n for n in ast.walk(node)
                if isinstance(n, ast.AnnAssign)
            ]
            assert len(column_assigns) == 3, (
                f"PlatformIdentity must have exactly 3 columns, found {len(column_assigns)}"
            )
            break

def test_manifest_swap_minimal_domain() -> None:
    """
    Swap personal_finance manifest for a minimal test domain.
    Platform must boot and generate tables without code changes.
    """
    import json
    import tempfile
    from pathlib import Path
    from sqlalchemy import MetaData

    # Create a minimal test domain manifest in a temp dir
    tmp_dir = Path(tempfile.mkdtemp())
    schema_dir = tmp_dir / "chests"
    schema_dir.mkdir()

    # Minimal flat schema
    (schema_dir / "items.schema.json").write_text(json.dumps({
        "$schema": "https://json-schema.org/draft/2020-12/schema",
        "type": "object",
        "properties": {
            "name": {"type": "string"},
            "score": {"type": "number"},
        },
        "required": ["name"],
        "additionalProperties": False,
    }))

    # Minimal manifest YAML
    import yaml  # type: ignore[import]
    manifest_data = {
        "domain": {"id": "test_domain", "name": "Test Domain", "version": "1.0.0", "locale_default": "en"},
        "chests": [{"id": "items", "type": "items", "scope": "domain",
                    "access": ["searchable", "tool_callable"],
                    "storage": {"encrypted": False, "localized": False, "ttl_days": None},
                    "schema_ref": "chests/items.schema.json"}],
    }
    manifest_path = tmp_dir / "manifest.yaml"
    manifest_path.write_text(yaml.dump(manifest_data))

    from app.domain.loader import load_manifest
    from app.data.factory import DomainModelFactory

    manifest = load_manifest(manifest_path)
    assert manifest.domain.id == "test_domain"

    factory = DomainModelFactory(MetaData())
    tables = factory.build_tables(manifest, tmp_dir, lambda: "key")
    assert "items" in tables
    assert tables["items"].name == "test_domain__items"

    # Verify columns: standard + name + score
    col_names = {c.name for c in tables["items"].columns}
    assert "name" in col_names
    assert "score" in col_names
    assert "identity_id" in col_names

    print("✓ Manifest swap test: minimal test_domain loaded and tables generated")

def test_manifest_loads_successfully() -> None:

def test_domain_model_factory_builds_all_tables() -> None:
    """DomainModelFactory generates 7 typed tables from personal_finance manifest."""
    from sqlalchemy import MetaData
    from app.domain.loader import load_manifest
    from app.data.factory import DomainModelFactory

    p = Path("domains/personal-finance/manifest.yaml")
    m = load_manifest(p)
    tables = DomainModelFactory(MetaData()).build_tables(m, p.parent, lambda: "k")
    assert len(tables) == 7
    assert "personal_finance__transactions" in {t.name for t in tables.values()}
    assert "personal_finance__user_prefs" in {t.name for t in tables.values()}


def test_tool_registry_generates_tools() -> None:
    """ToolRegistry generates LLM tools for all tool_callable + searchable chests."""
    from app.domain.loader import load_manifest
    from app.tools.registry import ToolRegistry

    p = Path("domains/personal-finance/manifest.yaml")
    m = load_manifest(p)

    assert any(n.startswith("search_") for n in names)
