"""
PersonalFinanceAuthPlugin — domain auth plugin for personal_finance.

Implements authenticate_request hookspec.
Validates JWT access tokens issued by personal_finance domain auth endpoints.
Returns platform identity_id on success, None on failure.

FIN-09: full v1.0 auth parity (email/password + Google OAuth + JWT rotation)
        implemented as domain plugin, not platform code.
CORE-09: domain plugin owns all auth logic; platform never inspects token.
"""

from __future__ import annotations

import logging
from typing import Any

from app.platform.hookspecs import hookimpl

logger = logging.getLogger(__name__)


class PersonalFinanceAuthPlugin:
    """
    Auth plugin for the personal_finance domain.

    Validates Bearer JWT tokens issued at login/OAuth.
    Maps valid tokens to platform_identities.id.
    """

    @hookimpl
    def authenticate_request(self, token: str) -> str | None:
        """
        Validate a JWT access token and return the identity_id.

        Returns None if token is invalid, expired, or not for this domain.
        """
        try:
            from app.core.security import decode_access_token  # noqa: PLC0415
            payload = decode_access_token(token)
            if not payload:
                return None
            # Token sub contains the platform identity_id for v2.0,
            # or user_id for v1.0 backwards compat
            identity_id = payload.get("sub")
            return identity_id
        except Exception as exc:
            logger.debug("PersonalFinanceAuthPlugin: auth failed: %s", exc)
            return None

    @hookimpl
    def categorize_transactions(self, transactions: list[dict]) -> list[dict]:
        """Delegate to existing categorization service."""
        # Thin wrapper — actual logic in Phase 39-02 Task 2
        return transactions

    @hookimpl
    def process_extraction(self, doc_type: str, raw_data: dict) -> dict:
        """personal_finance-specific post-processing of extracted documents."""
        # Phase 39: add payroll extraction handling (FIN-04)
        return raw_data
```

"""
PersonalFinanceCategoryPlugin — transaction categorization for personal_finance.

FIN-03: finance categorization runs as domain plugin.
FIN-04: Italian payroll extraction runs as personal_finance domain service.
"""

from __future__ import annotations

import logging
from typing import Any

from app.platform.hookspecs import hookimpl

logger = logging.getLogger(__name__)


class PersonalFinanceCategoryPlugin:
    """Wraps existing categorization_service as a domain plugin hookimpl."""

    @hookimpl
    def categorize_transactions(self, transactions: list[dict]) -> list[dict]:
        """
        Enrich transactions with finance-specific categories.
        Delegates to existing categorization_service logic.
        FIN-03: runs as domain plugin, not platform code.
        """
        try:
            from app.services.categorization_service import categorize_batch  # noqa: PLC0415
            return categorize_batch(transactions)
        except Exception as exc:
            logger.warning("CategoryPlugin: categorization failed: %s", exc)
            return transactions

    @hookimpl
    def process_extraction(self, doc_type: str, raw_data: dict) -> dict:
        """
        Post-process payslip extraction using Italian payroll parser.
        FIN-04: Italian payroll runs as personal_finance domain service.
        """
        if doc_type == "payslip":
            try:
                from app.services.italian_payroll import enrich_payslip  # noqa: PLC0415
                return enrich_payslip(raw_data)
            except Exception as exc:
                logger.warning("CategoryPlugin: payroll enrichment failed: %s", exc)
        return raw_data
```
