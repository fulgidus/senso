```python
"""
Schema finder for domain manifests.
Handles both kebab-case and snake_case file patterns for backward compatibility.
"""

import os
from pathlib import Path
from typing import Dict, List, Optional, Tuple

from app.domain.models import ChestDef


def find_chest_files(chest: ChestDef, base_dir: Path) -> Dict[str, Path]:
    """
    Find all files associated with a chest based on its ID.
    
    Supports both kebab-case and snake_case formats for backward compatibility.
    Prioritizes kebab-case files when both formats exist.
    
    Returns a dictionary mapping file types to paths:
    {
        'schema': Path to schema file,
        'tool': Path to tool file (if exists),
        'validator': Path to validator file (if exists),
        ...
    }
    """
    result: Dict[str, Path] = {}
    
    # First try kebab-case
    chest_id_kebab = chest.id.replace("_", "-")
    
    # Try in /chests/[chest-id]/ directory first (new structure)
    chest_dir = base_dir / "chests" / chest_id_kebab
    if chest_dir.exists():
        # Look for schema, tool, and other files
        for file_type, pattern in [
            ("schema", f"{chest_id_kebab}.schema.json"),
            ("tool", f"{chest_id_kebab}.tool.yaml"),
            ("validator", f"{chest_id_kebab}.validator.py"),
        ]:
            file_path = chest_dir / pattern
            if file_path.exists():
                result[file_type] = file_path
    
    # If not found in chest directory, check legacy patterns
    if "schema" not in result and chest.schema_ref:
        # Try direct reference from manifest first
        ref_path = base_dir / chest.schema_ref
        if ref_path.exists():
            result["schema"] = ref_path
        
        # Then try kebab-case in chests directory
        if "schema" not in result:
            legacy_path = base_dir / "chests" / f"{chest_id_kebab}.schema.json"
            if legacy_path.exists():
                result["schema"] = legacy_path
                
            # Finally try snake_case (oldest format)
            if "schema" not in result:
                snake_path = base_dir / "chests" / f"{chest.id}.schema.json"
                if snake_path.exists():
                    result["schema"] = snake_path
    
    # Check for tool file in chests directory (legacy)
    if "tool" not in result:
        legacy_tool = base_dir / "chests" / f"{chest_id_kebab}.tool.yaml"
        if legacy_tool.exists():
            result["tool"] = legacy_tool
            
        # Try snake_case tool (oldest format)
        if "tool" not in result:
            snake_tool = base_dir / "chests" / f"{chest.id}.tool.yaml"
            if snake_tool.exists():
                result["tool"] = snake_tool
    
    return result


def ensure_chest_directories(domain_dir: Path) -> None:
    """
    Ensure the chest directory structure exists for a domain.
    """
    chests_dir = domain_dir / "chests"
    if not chests_dir.exists():
        chests_dir.mkdir(parents=True, exist_ok=True)
```

Update `be/app/domain/tools.py`:
```python
"""
Domain-specific tool generation.

Tools are defined per chest and loaded from dedicated tool files.
"""

import os
from pathlib import Path
from typing import Dict, List, Optional, Any

import yaml
from pydantic import BaseModel, Field

from app.domain.models import ChestDef, DomainManifest
from app.domain.schema_finder import find_chest_files


class ToolDefinition(BaseModel):
    """Tool definition from tool.yaml file."""
    name: str
    description: str
    parameters: Dict[str, Any] = Field(default_factory=dict)
    response_schema: Optional[Dict[str, Any]] = None
    examples: List[Dict[str, Any]] = Field(default_factory=list)
    enabled: bool = True


def load_chest_tool(chest: ChestDef, base_dir: Path) -> Optional[ToolDefinition]:
    """
    Load tool definition for a specific chest from its tool file.
    
    Args:
        chest: The chest definition
        base_dir: Base directory of the domain
        
    Returns:
        ToolDefinition if found, None otherwise
    """
    chest_files = find_chest_files(chest, base_dir)
    if "tool" not in chest_files:
        return None
        
    tool_path = chest_files["tool"]
    try:
        with open(tool_path, "r") as f:
            tool_data = yaml.safe_load(f)
            return ToolDefinition(**tool_data)
    except Exception as e:
        # Log error and return None
        print(f"Error loading tool for chest {chest.id}: {e}")
        return None


def generate_tools_for_domain(manifest: DomainManifest, base_dir: Path) -> List[Dict[str, Any]]:
    """
    Generate tools for all chests in a domain that have tool files.
    
    Args:
        manifest: Domain manifest
        base_dir: Base directory of the domain
        
    Returns:
        List of tool definitions in OpenAI format
    """
    tools = []
    
    for chest in manifest.chests:
        if "tool_callable" not in chest.access:
            continue
            
        tool_def = load_chest_tool(chest, base_dir)
        if tool_def and tool_def.enabled:
            tool = {
                "type": "function",
                "function": {
                    "name": tool_def.name,
                    "description": tool_def.description,
                    "parameters": tool_def.parameters
                }
            }
            
            if tool_def.response_schema:
                tool["function"]["response_schema"] = tool_def.response_schema
                
            tools.append(tool)
    
    return tools
```
