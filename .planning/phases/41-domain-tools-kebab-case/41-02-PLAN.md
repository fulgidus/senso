# Plan 41-02: Tool Registry and Parallel Tool Execution

## Goal

Implement a tool registry system that lists all available tools to the LLM and enables parallel/batch tool execution rather than forcing the LLM to loop over sequential tool calls. This will improve efficiency and allow for more natural reasoning.

## must_haves

- Create a central tool registry that collects tools across all domains
- Enable the LLM to see all available tools in a single context
- Support parallel tool execution in a single request
- Allow batch tool calls in a single response
- Update the coaching service to handle parallel tool execution
- Maintain compatibility with the LLM's function calling capabilities
- Add tests to verify the functionality

"""
Tool registry for the platform.

Collects and manages tools from all domains, making them available to the LLM
for parallel execution.
"""

from pathlib import Path
from typing import Dict, List, Any, Optional, Set

from app.domain.loader import load_manifest
from app.domain.tools import generate_tools_for_domain
from app.domain.models import DomainManifest


class ToolRegistry:
    """Registry for all tools available to the LLM."""
    
    def __init__(self):
        self.domains: Dict[str, DomainManifest] = {}
        self.domain_paths: Dict[str, Path] = {}
        self.tools_by_domain: Dict[str, List[Dict[str, Any]]] = {}
        self.tools_by_name: Dict[str, Dict[str, Any]] = {}
    
    def register_domain(self, domain_id: str, manifest_path: Path) -> None:
        """
        Register a domain and load its tools.
        
        Args:
            domain_id: Domain identifier
            manifest_path: Path to the domain manifest file
        """
        manifest = load_manifest(manifest_path)
        self.domains[domain_id] = manifest
        self.domain_paths[domain_id] = manifest_path.parent
        
        # Generate tools for the domain
        domain_tools = generate_tools_for_domain(manifest, manifest_path.parent)
        self.tools_by_domain[domain_id] = domain_tools
        
        # Register tools by name for quick lookup
        for tool in domain_tools:
            tool_name = tool["function"]["name"]
            # Store domain id with the tool for context
            tool["domain_id"] = domain_id
            self.tools_by_name[tool_name] = tool
    
    def get_all_tools(self) -> List[Dict[str, Any]]:
        """
        Get all registered tools across domains.
        
        Returns:
            List of tools in OpenAI format
        """
        all_tools = []
        for domain_tools in self.tools_by_domain.values():
            all_tools.extend(domain_tools)
        return all_tools
    
    def get_tools_for_domain(self, domain_id: str) -> List[Dict[str, Any]]:
        """
        Get tools for a specific domain.
        
        Args:
            domain_id: Domain identifier
            
        Returns:
            List of tools in OpenAI format, or empty list if domain not registered
        """
        return self.tools_by_domain.get(domain_id, [])
    
    def get_tool_by_name(self, tool_name: str) -> Optional[Dict[str, Any]]:
        """
        Get a tool by its name.
        
        Args:
            tool_name: Name of the tool
            
        Returns:
            Tool definition or None if not found
        """
        return self.tools_by_name.get(tool_name)
    
    def reload_domain(self, domain_id: str) -> None:
        """
        Reload tools for a domain.
        
        Args:
            domain_id: Domain identifier
        """
        if domain_id in self.domains and domain_id in self.domain_paths:
            manifest = self.domains[domain_id]
            domain_path = self.domain_paths[domain_id]
            
            # Remove existing tools for this domain
            domain_tools = self.tools_by_domain.get(domain_id, [])
            for tool in domain_tools:
                tool_name = tool["function"]["name"]
                if tool_name in self.tools_by_name:
                    del self.tools_by_name[tool_name]
            
            # Regenerate tools
            domain_tools = generate_tools_for_domain(manifest, domain_path)
            self.tools_by_domain[domain_id] = domain_tools
            
            # Re-register tools by name
            for tool in domain_tools:
                tool_name = tool["function"]["name"]
                tool["domain_id"] = domain_id
                self.tools_by_name[tool_name] = tool
    
    def get_registered_domains(self) -> Set[str]:
        """
        Get set of registered domain IDs.
        
        Returns:
            Set of domain identifiers
        """
        return set(self.domains.keys())


# Global registry instance
_registry = ToolRegistry()


def get_tool_registry() -> ToolRegistry:
    """
    Get the global tool registry.
    
    Returns:
        Global ToolRegistry instance
    """
    return _registry


def register_domain(domain_id: str, manifest_path: Path) -> None:
    """
    Register a domain with the global registry.
    
    Args:
        domain_id: Domain identifier
        manifest_path: Path to the domain manifest file
    """
    _registry.register_domain(domain_id, manifest_path)


def get_all_tools() -> List[Dict[str, Any]]:
    """
    Get all tools from the global registry.
    
    Returns:
        List of all registered tools
    """
    return _registry.get_all_tools()


def get_tools_for_domain(domain_id: str) -> List[Dict[str, Any]]:
    """
    Get tools for a specific domain from the global registry.
    
    Args:
        domain_id: Domain identifier
        
    Returns:
        List of tools for the domain
    """
    return _registry.get_tools_for_domain(domain_id)

"""
LLM API wrapper with support for parallel tool execution.

Provides a unified interface to different LLM providers with support for
parallel/batch tool calls.
"""

import asyncio
from typing import Dict, List, Any, Optional, Callable, Union

from app.domain.tool_registry import get_tools_for_domain, get_all_tools, get_tool_by_name


class ParallelToolHandler:
    """Handler for parallel tool execution."""
    
    def __init__(self, domain_id: str = None, tool_implementations: Dict[str, Callable] = None):
        """
        Initialize the handler.
        
        Args:
            domain_id: Domain identifier to limit tool scope, or None for all domains
            tool_implementations: Dict mapping tool names to their implementation functions
        """
        self.domain_id = domain_id
        self.tool_implementations = tool_implementations or {}
        
        # Load tools from registry
        if domain_id:
            self.tools = get_tools_for_domain(domain_id)
        else:
            self.tools = get_all_tools()
    
    def register_tool_implementation(self, tool_name: str, implementation: Callable) -> None:
        """
        Register implementation for a tool.
        
        Args:
            tool_name: Name of the tool
            implementation: Function implementing the tool
        """
        self.tool_implementations[tool_name] = implementation
    
    async def execute_tool(self, tool_name: str, arguments: Dict[str, Any]) -> Any:
        """
        Execute a tool asynchronously.
        
        Args:
            tool_name: Name of the tool to execute
            arguments: Arguments to pass to the tool
            
        Returns:
            Tool execution result
            
        Raises:
            ValueError: If tool not found or implementation not registered
        """
        if tool_name not in self.tool_implementations:
            raise ValueError(f"No implementation registered for tool {tool_name}")
            
        # Get the implementation
        implementation = self.tool_implementations[tool_name]
        
        # Execute the tool (might be sync or async)
        if asyncio.iscoroutinefunction(implementation):
            result = await implementation(**arguments)
        else:
            # Run sync function in thread pool
            loop = asyncio.get_event_loop()
            result = await loop.run_in_executor(
                None,
                lambda: implementation(**arguments)
            )
        
        return result
    
    async def execute_parallel_tools(self, tool_calls: List[Dict[str, Any]]) -> List[Any]:
        """
        Execute multiple tools in parallel.
        
        Args:
            tool_calls: List of tool calls, each with "name" and "arguments"
            
        Returns:
            List of results in the same order as the tool calls
        """
        tasks = []
        for call in tool_calls:
            tool_name = call["name"]
            arguments = call["arguments"]
            task = self.execute_tool(tool_name, arguments)
            tasks.append(task)
        
        # Execute all tools in parallel
        if tasks:
            results = await asyncio.gather(*tasks, return_exceptions=True)
        else:
            results = []
        
        # Convert exceptions to error messages
        processed_results = []
        for result in results:
            if isinstance(result, Exception):
                processed_results.append({"error": str(result)})
            else:
                processed_results.append(result)
        
        return processed_results


class LLMClient:
    """Client for interacting with LLMs with parallel tool support."""
    
    def __init__(self, domain_id: str, tool_handler: ParallelToolHandler = None):
        """
        Initialize the LLM client.
        
        Args:
            domain_id: Domain identifier
            tool_handler: Handler for tool execution, or None to create one
        """
        self.domain_id = domain_id
        self.tool_handler = tool_handler or ParallelToolHandler(domain_id=domain_id)
    
    async def complete_with_tools(
        self,
        messages: List[Dict[str, Any]],
        model: str,
        temperature: float = 0.7,
        max_tokens: int = 1000
    ) -> Dict[str, Any]:
        """
        Complete a conversation with tool support.
        
        Args:
            messages: Conversation messages
            model: LLM model to use
            temperature: Sampling temperature
            max_tokens: Maximum tokens in response
            
        Returns:
            LLM response with tool calls if any
        """

            "id": "chat-123",
            "model": model,
            "choices": [
                {
                    "message": {
                        "role": "assistant",
