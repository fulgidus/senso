# Plan 38-01: SDUI-Lite Card Registry + Domain UI API

## Goal

Implement a domain-driven card rendering system for React: backend exposes `/domain/ui` with manifest UI descriptors (card types, theme, voice config). Frontend `CardRegistry` maps card type strings to React components — replacing hardcoded `AffordabilityVerdictCard` etc. in ChatScreen. Persona theming and voice provider config loaded from manifest.

## must_haves

- `GET /domain/ui` returns domain UI descriptor: card types, persona themes, voice config
- `CardRegistry` in frontend maps `card_type` string → React component (registered at app init)
- `ChatScreen.tsx` renders coaching cards via `CardRegistry.render(card)` — no hardcoded component names
- New card types in manifest = new CardRegistry entry = renders automatically (no ChatScreen changes)
- Persona theming (CSS vars: primary color, avatar URL, label) loaded from manifest at startup
- Voice I/O provider (TTS/STT) config from manifest (`FE-05`)
- `GET /domain/ui` returns 200 with `card_types`, `theme`, `voice` sections
- `pnpm --prefix fe build` exits 0 after changes

```python
"""
Domain API — exposes domain manifest UI descriptors to the frontend.

FE-01/02/03/06: card types, profile views, persona theming from manifest.
FE-05: voice I/O provider config.
"""

from __future__ import annotations

from fastapi import APIRouter, Request
from fastapi.responses import JSONResponse

router = APIRouter(prefix="/domain", tags=["domain"])


@router.get("/ui")
async def get_domain_ui(request: Request) -> JSONResponse:
    """
    Return domain UI descriptor for frontend card registry and theming.

    Response shape:
    {
      "domain_id": "personal_finance",
      "card_types": [{"type": "affordability_card", "renderer": "sdui", "config": {}}],
      "personas": [{"id": "mentore-saggio", "name": "...", "theme": {...}}],
      "voice": {"tts_provider": "elevenlabs", "stt_provider": "webSpeech", ...},
      "locale_default": "it"
    }
    """
    manifest = request.app.state.manifest

    # Card types from manifest UI descriptors
    card_types = [
        {"type": c.type, "renderer": c.renderer, "config": dict(c.config)}
        for c in manifest.ui.components
    ]

    # Persona themes from personas config
    persona_themes: list[dict] = []
    if hasattr(request.app.state, "behavior_layer"):
        pass  # behavior_layer doesn't have persona data
    # Load from persona engine if available
    if hasattr(request.app.state, "persona_engine"):
        engine = request.app.state.persona_engine
        for persona in engine.list_personas():
            persona_themes.append({
                "id": persona.get("id"),
                "name": persona.get("name"),
                "theme": persona.get("theme", {}),
                "avatar_url": persona.get("avatarUrl"),
            })
    else:
        # Fallback: load from personas/config.json directly
        from app.personas.loader import _load_config  # noqa: PLC0415
        config = _load_config()
        for persona in config.get("personas", []):
            if persona.get("available", True):
                persona_themes.append({
                    "id": persona.get("id"),
                    "name": persona.get("name"),
                    "theme": persona.get("theme", {}),
                    "avatar_url": persona.get("avatarUrl"),
                })

    # Voice config from manifest auth / settings
    from app.core.config import get_settings  # noqa: PLC0415
    settings = get_settings()
    voice_config = {
        "tts_enabled": settings.tts_enabled,
        "stt_provider": settings.stt_provider,
        "tts_provider": "elevenlabs" if settings.tts_enabled else "browser",
    }

    return JSONResponse({
        "domain_id": manifest.domain.id,
        "domain_name": manifest.domain.name,
        "locale_default": manifest.domain.locale_default,
        "card_types": card_types,
        "personas": persona_themes,
        "voice": voice_config,
    })
```

```tsx
/**
 * CardRegistry — domain-driven card renderer.
 *
 * Maps card type strings (from domain manifest + coaching response) to React components.
 * New domains register their card types at app init; ChatScreen renders via registry.
 *
 * FE-03: coaching response cards render via SDUI — no hardcoded card type names in ChatScreen.
 */

import React from "react"

type CardRenderer = (props: { card: Record<string, unknown>; locale: string }) => React.ReactElement | null

const _registry = new Map<string, CardRenderer>()

/** Register a React component for a card type. */
export function registerCard(type: string, renderer: CardRenderer): void {
  _registry.set(type, renderer)
}

/** Render a card by its type. Returns null if no renderer registered. */
export function renderCard(
  type: string,
  card: Record<string, unknown>,
  locale: string,
): React.ReactElement | null {
  const Renderer = _registry.get(type)
  if (!Renderer) return null
  return <Renderer card={card} locale={locale} />
}

/** Check if a card type is registered. */
export function hasCard(type: string): boolean {
  return _registry.has(type)
}

/** Return all registered card type keys. */
export function registeredTypes(): string[] {
  return Array.from(_registry.keys())
}
```

```ts
/**
 * useDomainUI — fetches domain UI descriptor and applies persona theming.
 * FE-06: persona theming (colors, avatar, labels) from domain manifest.
 */
import { useEffect, useState } from "react"
import { apiFetch } from "../api/client"

interface DomainUIConfig {
  domain_id: string
  domain_name: string
  locale_default: string
  card_types: Array<{ type: string; renderer: string; config: Record<string, unknown> }>
  personas: Array<{ id: string; name: string; theme: Record<string, string>; avatar_url?: string }>
  voice: { tts_enabled: boolean; stt_provider: string; tts_provider: string }
}

export function useDomainUI(): DomainUIConfig | null {
  const [config, setConfig] = useState<DomainUIConfig | null>(null)

  useEffect(() => {
    apiFetch<DomainUIConfig>("/domain/ui")
      .then((data) => {
        setConfig(data)
        // Apply persona theme CSS vars if theme data present
        for (const persona of data.personas) {
          if (persona.theme) {
            for (const [key, val] of Object.entries(persona.theme)) {
              document.documentElement.style.setProperty(
                `--persona-${persona.id}-${key}`, val
              )
            }
          }
        }
      })
      .catch((err) => {
        console.warn("useDomainUI: failed to load domain UI config", err)
      })
  }, [])

  return config
}
```
