```python
"""
ChestRouter — routes extracted documents to domain Chests via ChestAccessor.

Replaces the hardcoded TransactionModel.insert() in ingestion_service.py.
Uses manifest extractor declarations to find target chests for each document_type.
"""

from __future__ import annotations

import logging
from typing import Any

from sqlalchemy.orm import Session

from app.data.accessor import ChestAccessor
from app.data.manager import DomainManager
from app.domain.manifest import DomainManifest
from app.domain.registry import ChestRegistry
from app.schemas.ingestion import ExtractedDocument

logger = logging.getLogger(__name__)


class ChestRouter:
    """
    Maps document_type → target chest_ids using manifest extractor declarations.

    For each extractor in manifest.extractors:
        extractor.document_types → extractor.target_chests (list of chest_ids)

    Falls back to writing transactions to the 'transactions' chest and
    document metadata to the 'user_profile' chest when no explicit mapping exists
    (preserves v1.0 behaviour as default for personal_finance domain).
    """

    # Default routing when no extractor declares document_type explicitly
    _DEFAULT_TRANSACTION_CHEST = "transactions"
    _DEFAULT_PROFILE_CHEST = "user_profile"

    def __init__(
        self,
        manifest: DomainManifest,
        manager: DomainManager,
        registry: ChestRegistry,
    ) -> None:
        self._manifest = manifest
        self._manager = manager
        self._registry = registry
        # Build lookup: document_type → list[chest_id]
        self._routing: dict[str, list[str]] = {}
        for extractor in manifest.extractors:
            for doc_type in extractor.document_types:
                self._routing.setdefault(doc_type, []).extend(extractor.target_chests)

    def route(
        self,
        doc: ExtractedDocument,
        identity_id: str,
        domain_id: str,
        db: Session,
        locale: str = "it",
    ) -> list[str]:
        """
        Route an extracted document to its target Chests.

        Returns list of created row IDs.
        """
        created_ids: list[str] = []
        target_chests = self._routing.get(doc.document_type, [])

        if not target_chests:
            # Fallback: transactions go to transactions chest, rest to user_profile
            target_chests = self._default_targets(doc)

        for chest_id in target_chests:
            try:
                chest_def = self._registry.get(chest_id)
                table = self._manager.get_table(domain_id, chest_id)
                accessor = ChestAccessor(table, chest_def)
                rows = self._doc_to_rows(doc, chest_id)
                for row_data in rows:
                    row_id = accessor.create(
                        db,
                        data=row_data,
                        identity_id=identity_id,
                        domain_id=domain_id,
                        locale=locale,
                    )
                    created_ids.append(row_id)
                    logger.debug(
                        "ChestRouter: wrote to %s/%s id=%s",
                        domain_id, chest_id, row_id,
                    )
            except KeyError:
                logger.warning(
                    "ChestRouter: chest '%s' not found for doc_type '%s' — skipping",
                    chest_id, doc.document_type,
                )
            except Exception as exc:
                logger.error(
                    "ChestRouter: failed to write to chest '%s': %s",
                    chest_id, exc,
                )

        return created_ids

    def _default_targets(self, doc: ExtractedDocument) -> list[str]:
        """Fallback routing when no extractor declares this document_type."""
        targets: list[str] = []
        if doc.transactions:
            targets.append(self._DEFAULT_TRANSACTION_CHEST)
        # Non-ledger documents with profile-level data
        if hasattr(doc, "income_summary") or hasattr(doc, "profile_fields"):
            targets.append(self._DEFAULT_PROFILE_CHEST)
        if not targets:
            targets.append(self._DEFAULT_TRANSACTION_CHEST)
        return targets

    def _doc_to_rows(
        self, doc: ExtractedDocument, chest_id: str
    ) -> list[dict[str, Any]]:
        """Convert an ExtractedDocument to row dicts for the target chest."""
        if chest_id == self._DEFAULT_TRANSACTION_CHEST:
            # Each transaction becomes one row
            rows = []
            for txn in (doc.transactions or []):
                row = txn if isinstance(txn, dict) else txn.model_dump()
                row["source_document_type"] = doc.document_type
                rows.append(row)
            return rows
        # For other chests: write the whole doc as one row
        doc_dict = doc.model_dump() if hasattr(doc, "model_dump") else dict(doc)
        return [doc_dict]
```
