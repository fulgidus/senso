```python
"""
ToolGenerator — derives OpenAI function-calling tool dicts from ChestDef metadata.

Rules:
  tool_callable chest  → get_{id}  (read all rows for current user)
                       → update_{id} (write/merge fields)
  searchable chest     → search_{id} (BM25 query)
  tool_spec override   → use verbatim (escape hatch for complex tools)

Zero finance-specific logic. Any domain using Chest metadata gets tools for free.
"""

from __future__ import annotations

import logging
from typing import Any

from app.domain.manifest import ChestDef

logger = logging.getLogger(__name__)

# ── JSON Schema type → OpenAI parameter type ─────────────────────────────────

_JSON_TO_OAI: dict[str, str] = {
    "string": "string",
    "number": "number",
    "integer": "integer",
    "boolean": "boolean",
}

# Platform columns — excluded from update tool parameters
_PLATFORM_COLS = frozenset(
    {"id", "identity_id", "domain_id", "locale", "created_at", "updated_at", "expires_at"}
)


def _schema_to_oai_properties(schema: dict[str, Any]) -> tuple[dict, list[str]]:
    """Convert JSON Schema properties to OpenAI parameter properties + required list."""
    properties: dict[str, Any] = {}
    required: list[str] = []
    for prop_name, prop_schema in schema.get("properties", {}).items():
        if prop_name in _PLATFORM_COLS:
            continue
        json_type = prop_schema.get("type", "string")
        oai_type = _JSON_TO_OAI.get(json_type, "string")
        entry: dict[str, Any] = {"type": oai_type}
        if "description" in prop_schema:
            entry["description"] = prop_schema["description"]
        elif "title" in prop_schema:
            entry["description"] = prop_schema["title"]
        properties[prop_name] = entry
        if prop_name in schema.get("required", []):
            required.append(prop_name)
    return properties, required


class ToolGenerator:
    """Generates OpenAI-compatible tool dicts from ChestDef + JSON Schema."""

    def generate(
        self,
        chest: ChestDef,
        schema: dict[str, Any],
        locale_default: str = "it",
    ) -> list[dict[str, Any]]:
        """
        Return 0–3 tool dicts for this chest based on its access modes.

        If chest.tool_spec is set, return [chest.tool_spec] verbatim.
        """
        if chest.tool_spec:
            logger.debug("Chest '%s': using tool_spec override", chest.id)
            return [chest.tool_spec]

        tools: list[dict[str, Any]] = []

        if "tool_callable" in chest.access:
            tools.append(self._get_tool(chest, schema))
            tools.append(self._update_tool(chest, schema))

        if "searchable" in chest.access:
            tools.append(self._search_tool(chest, locale_default))

        if not tools:
            logger.debug("Chest '%s': no tool_callable/searchable access — no tools generated", chest.id)

        return tools

    def _get_tool(self, chest: ChestDef, schema: dict[str, Any]) -> dict[str, Any]:
        """Generate get_{chest_id} tool — reads all rows for the current user."""
        return {
            "type": "function",
            "function": {
                "name": f"get_{chest.id}",
                "description": (
                    f"Retrieve the user's {chest.id.replace('_', ' ')} data. "
                    f"Returns all records for the current user."
                ),
                "parameters": {
                    "type": "object",
                    "properties": {},
                    "required": [],
                    "additionalProperties": False,
                },
            },
        }

    def _update_tool(self, chest: ChestDef, schema: dict[str, Any]) -> dict[str, Any]:
        """Generate update_{chest_id} tool — writes/merges fields."""
        props, required = _schema_to_oai_properties(schema)
        return {
            "type": "function",
            "function": {
                "name": f"update_{chest.id}",
                "description": (
                    f"Update the user's {chest.id.replace('_', ' ')} record. "
                    f"Provide only the fields to change."
                ),
                "parameters": {
                    "type": "object",
                    "properties": props,
                    "required": [],  # all optional for update
                    "additionalProperties": False,
                },
            },
        }

    def _search_tool(self, chest: ChestDef, locale_default: str) -> dict[str, Any]:
        """Generate search_{chest_id} BM25 search tool."""
        return {
            "type": "function",
            "function": {
                "name": f"search_{chest.id}",
                "description": (
                    f"Search the {chest.id.replace('_', ' ')} catalog using keywords. "
                    f"Returns ranked results matching the query."
                ),
                "parameters": {
                    "type": "object",
                    "properties": {
                        "query": {
                            "type": "string",
                            "description": "Keywords describing what to search for.",
                        },
                        "locale": {
                            "type": "string",
                            "description": f"Language locale (default: {locale_default}).",
                        },
                        "top_k": {
                            "type": "integer",
                            "description": "Maximum results to return (default 5).",
                        },
                    },
                    "required": ["query", "locale"],
                    "additionalProperties": False,
                },
            },
        }
```

```python
"""
ToolRegistry — holds all LLM tool definitions and executor dispatch for a domain.

Populated at startup from ToolGenerator output.
Domain plugins (Phase 36/39) can register custom executors via register_executor().
CoachingService injects this registry to replace hardcoded tool list.
"""

from __future__ import annotations

import json
import logging
from pathlib import Path
from typing import Any, Callable

from app.domain.manifest import DomainManifest
from app.tools.generator import ToolGenerator

logger = logging.getLogger(__name__)


class ToolRegistry:
    """
    Domain-scoped registry of LLM tool definitions and their Python executors.

    tool_defs: list[dict] — OpenAI-format tool dicts passed to LLM
    executors: {name: fn} — Python callables invoked when LLM calls the tool
    """

    def __init__(self) -> None:
        self._tool_defs: list[dict[str, Any]] = []
        self._executors: dict[str, Callable] = {}

    # ── Build from manifest ───────────────────────────────────────────────────

    @classmethod
    def build(
        cls,
        manifest: DomainManifest,
        manifest_dir: Path,
    ) -> "ToolRegistry":
        """Build a ToolRegistry from all chests in a DomainManifest."""
        registry = cls()
        generator = ToolGenerator()

        for chest in manifest.chests:
            schema: dict[str, Any] = {}
            if chest.schema_ref:
                schema_path = manifest_dir / chest.schema_ref
                if schema_path.exists():
                    schema = json.loads(schema_path.read_text(encoding="utf-8"))

            tools = generator.generate(chest, schema, manifest.domain.locale_default)
            for tool in tools:
                registry._tool_defs.append(tool)
                tool_name = tool["function"]["name"]
                logger.debug("ToolRegistry: registered tool '%s' for chest '%s'", tool_name, chest.id)

        logger.info(
            "ToolRegistry: built %d tool(s) for domain '%s'",
            len(registry._tool_defs),
            manifest.domain.id,
        )
        return registry

    # ── Access ────────────────────────────────────────────────────────────────

    def get_all(self) -> list[dict[str, Any]]:
        """Return all tool dicts (OpenAI format) for injection into LLM calls."""
        return list(self._tool_defs)

    def tool_names(self) -> list[str]:
        return [t["function"]["name"] for t in self._tool_defs]

    # ── Executor binding ──────────────────────────────────────────────────────

    def register_executor(self, name: str, fn: Callable) -> None:
        """Bind a Python callable to a tool name. Called by domain plugins."""
        self._executors[name] = fn
        logger.debug("ToolRegistry: executor registered for '%s'", name)

    def get_executor(self, name: str) -> Callable | None:
        """Return the executor for a tool name, or None if not bound."""
        return self._executors.get(name)

    def has_executor(self, name: str) -> bool:
        return name in self._executors
```
