```python
"""
Webhook connector — receives, validates, and routes inbound domain data.

Endpoint: POST /webhook/{domain_id}/{chest_id}
Auth:     HMAC-SHA256 signature in X-Webhook-Signature header

Secret resolution order:
  1. env var WEBHOOK_SECRET_{DOMAIN_ID}_{CHEST_ID} (uppercased, hyphens→underscores)
  2. env var WEBHOOK_SECRET (global fallback)
  3. 401 if neither set

DATA-04: Webhook connector receives, validates HMAC, routes to Chests.
"""

from __future__ import annotations

import hashlib
import hmac
import json
import logging
import os
from typing import Any

from fastapi import APIRouter, HTTPException, Request, status
from fastapi.responses import JSONResponse

from app.data.accessor import ChestAccessor
from app.data.manager import DomainInactiveError

logger = logging.getLogger(__name__)

router = APIRouter(prefix="/webhook", tags=["webhook"])


def _get_webhook_secret(domain_id: str, chest_id: str) -> str | None:
    """Resolve webhook HMAC secret from environment variables."""
    # Specific secret: WEBHOOK_SECRET_PERSONAL_FINANCE_TRANSACTIONS
    specific_key = f"WEBHOOK_SECRET_{domain_id.upper()}_{chest_id.upper()}".replace("-", "_")
    secret = os.getenv(specific_key)
    if secret:
        return secret
    # Global fallback
    return os.getenv("WEBHOOK_SECRET")


def _verify_hmac(payload: bytes, secret: str, signature_header: str) -> bool:
    """Verify HMAC-SHA256 signature. Returns False on any mismatch."""
    try:
        received = signature_header.removeprefix("sha256=")
        expected = hmac.new(secret.encode(), payload, hashlib.sha256).hexdigest()
        return hmac.compare_digest(expected, received)
    except Exception:
        return False


@router.post("/{domain_id}/{chest_id}", status_code=status.HTTP_202_ACCEPTED)
async def receive_webhook(
    domain_id: str,
    chest_id: str,
    request: Request,
) -> JSONResponse:
    """
    Receive an inbound webhook payload and route to the target Chest.

    Headers required:
        X-Webhook-Signature: sha256=<hmac-sha256-hex>
        Content-Type: application/json
    """
    # 1. Read raw body (needed for HMAC before JSON parse)
    body = await request.body()

    # 2. Resolve and verify HMAC secret
    secret = _get_webhook_secret(domain_id, chest_id)
    if not secret:
        logger.warning(
            "Webhook: no secret configured for %s/%s — rejecting",
            domain_id, chest_id,
        )
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail={"code": "webhook_secret_not_configured"},
        )

    sig_header = request.headers.get("X-Webhook-Signature", "")
    if not _verify_hmac(body, secret, sig_header):
        logger.warning("Webhook: HMAC validation failed for %s/%s", domain_id, chest_id)
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail={"code": "invalid_signature"},
        )

    # 3. Parse JSON payload
    try:
        payload: dict[str, Any] = json.loads(body)
    except json.JSONDecodeError:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail={"code": "invalid_json"},
        )

    # 4. Route to Chest
    domain_manager = request.app.state.domain_manager
    chest_registry = request.app.state.chest_registry
    manifest = request.app.state.manifest

    # Validate domain matches active manifest
    if manifest.domain.id != domain_id:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail={"code": "domain_not_found", "domain_id": domain_id},
        )

    try:
        table = domain_manager.get_table(domain_id, chest_id)
    except DomainInactiveError:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail={"code": "domain_inactive", "domain_id": domain_id},
        )
    except KeyError:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail={"code": "chest_not_found", "chest_id": chest_id},
        )

    chest_def = chest_registry.get(chest_id)

    from app.db.session import SessionLocal  # noqa: PLC0415
    db = SessionLocal()
    try:
        accessor = ChestAccessor(table, chest_def)
        # Webhook payloads are domain-scoped (no identity_id) unless payload includes one
        identity_id = payload.pop("identity_id", None)
        row_id = accessor.create(
            db,
            data=payload,
            identity_id=identity_id,
            domain_id=domain_id,
            locale=payload.get("locale"),
        )
        db.commit()
    except Exception as exc:
        db.rollback()
        logger.error("Webhook: failed to write to chest %s/%s: %s", domain_id, chest_id, exc)
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail={"code": "write_failed"},
        )
    finally:
        db.close()

    logger.info("Webhook: wrote row %s to %s/%s", row_id, domain_id, chest_id)
    return JSONResponse(
        status_code=status.HTTP_202_ACCEPTED,
        content={"status": "accepted", "id": row_id},
    )
```
