```typescript
/**
 * Mock POST /coaching/stt — returns a fake transcript JSON response.
 * Simulates successful server-side STT transcription without a real microphone.
 *
 * @param transcript - The transcript text to return (default: Italian test phrase)
 */
export function mockCoachingSTT(
    page: Page,
    transcript = "Quanto posso spendere in affitto questo mese?"
): void {
    page.route("**/coaching/stt", (route: Route) => {
        route.fulfill({
            status: 200,
            contentType: "application/json",
            body: JSON.stringify({ text: transcript }),
        })
    })
}

/**
 * Mock POST /coaching/tts — returns fake audio bytes with audio/mpeg content type.
 * Simulates successful TTS audio generation without a real ElevenLabs key.
 *
 * Pass a real audio buffer (e.g., from fs.readFileSync) for tests that validate
 * Audio element playback; omit for tests that only check UI state changes.
 *
 * @param audioBuffer - Optional real audio bytes; defaults to minimal fake MP3 bytes
 */
export function mockCoachingTTS(page: Page, audioBuffer?: Buffer): void {
    // Minimal valid MP3 header bytes so Audio element doesn't throw "unsupported format"
    const fakeAudio = audioBuffer ?? Buffer.from(
        "ID3" + "\x03\x00\x00\x00\x00\x00\x00",  // ID3v2.3 header (11 bytes)
    )
    page.route("**/coaching/tts", (route: Route) => {
        route.fulfill({
            status: 200,
            contentType: "audio/mpeg",
            body: fakeAudio,
        })
    })
}
```

Note: `Buffer` is a Node.js global available in Playwright's Node context. No import needed.

```typescript
/**
 * Inject browser API mocks needed for voice mode to function without real hardware.
 * Must be called BEFORE page.goto() so scripts are injected on load.
 */
async function setupVoiceMocks(page: Parameters<typeof mockAllCoaching>[0]): Promise<void> {
    // Mock getUserMedia so toggleVoiceMode completes the permission prime without a dialog.
    // The stream is released immediately by toggleVoiceMode (the micStreamRef removal fix).
    await page.addInitScript(() => {
        const fakeTrack = {
            kind: "audio",
            stop: () => { /* no-op */ },
            enabled: true,
            id: "fake-track-id",
        } as unknown as MediaStreamTrack

        const fakeStream = {
            getTracks: () => [fakeTrack],
            getAudioTracks: () => [fakeTrack],
        } as unknown as MediaStream

        Object.defineProperty(navigator, "mediaDevices", {
            value: {
                getUserMedia: () => Promise.resolve(fakeStream),
                enumerateDevices: () => Promise.resolve([]),
            },
            writable: true,
            configurable: true,
        })
    })

    // Mock Web Speech API so the app thinks STT is available and the mic button is enabled.
    // The mock recognition never fires results — tests that need transcripts use the
    // MediaRecorder path or page.route() to inject the response.
    await page.addInitScript(() => {
        class MockSpeechRecognition extends EventTarget {
            lang = ""
            continuous = false
            interimResults = false
            maxAlternatives = 1
            onresult: ((e: Event) => void) | null = null
            onend: (() => void) | null = null
            onerror: ((e: Event) => void) | null = null
            start() { /* no-op: test controls when onend fires */ }
            stop() { this.onend?.() }
            abort() { this.onend?.() }
        }
        ;(window as Window & { SpeechRecognition?: typeof MockSpeechRecognition })
            .SpeechRecognition = MockSpeechRecognition
    })
}
```

## Risks

- **MediaRecorder in JSDOM** (Task 4): jsdom does not implement MediaRecorder. The test mocks `window.MediaRecorder` entirely. If `detectBackend()` runs before the mock is set (before the `useEffect` with `setBackend(detectBackend())`), the test may see `backend = "none"`. Use `act(() => { await Promise.resolve() })` to flush the effect. If tests still fail due to timing, use `waitFor` from `@testing-library/react`.
- **Playwright mediaDevices mock** (Task 3): `page.addInitScript()` scripts run in the frame context before the page loads. However, if `navigator.mediaDevices` is a non-writable descriptor, `Object.defineProperty` may throw. Use `configurable: true, writable: true` in the defineProperty call. Verify the mock works by checking the console output.
- **Web Speech API mock in Playwright** (Task 3): The mock SpeechRecognition class dispatches `onerror` after `start()`. If the voice mode component checks for `SpeechRecognition` availability via `typeof window.SpeechRecognition !== 'undefined'` at mount (not at call time), the script injection order matters. Since `addInitScript` runs before page scripts, this should be safe.
- **`pnpm test --run` vs `pnpm test`**: Use `--run` flag to run tests once without watch mode (required for CI and Docker).
- **Audio fixture fallback**: If ffmpeg is not available and the Python fallback script is used, verify the generated MP3 files are non-zero with `wc -c`. Playwright's mockCoachingTTS does not use the fixture files directly (it uses fake bytes), so fixture file quality only matters for tests that explicitly read them from disk.
- **Playwright test selector fragility**: Tests use `aria-label` and class-based selectors. If i18n locale in the test environment is `it` (not `en`), aria-labels will be Italian. Use both English and Italian regex patterns (e.g., `/Start voice recording|Avvia/`) to handle both locales.
