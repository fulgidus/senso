    ```ts
    // senso/src/api/adminMerchantApi.ts
    import { apiRequest } from "@/api/api-client";
    
    const API_BASE = import.meta.env.VITE_API_URL ?? "http://localhost:8000";
    
    export interface MerchantMapAdminDTO {
      id: string; description_raw: string; canonical_merchant: string | null;
      category: string; confidence: number; learned_method: string;
      learned_provider_model: string | null; learned_at: string;
      contributing_user_obfuscated: string | null;
      is_blacklisted: boolean; blacklisted_reason: string | null;
    }
    
    export interface ModerationLogAdminDTO {
      id: string; user_id: string; content_type: string; content_ref_id: string | null;
      detected_violations: string[]; severity: string; action_taken: string; created_at: string;
    }
    
    export async function getMerchantMap(params: { search?: string; method?: string; blacklisted?: string; limit?: number; offset?: number }): Promise<MerchantMapAdminDTO[]> {
      const q = new URLSearchParams();
      if (params.search) q.set("search", params.search);
      if (params.method) q.set("method", params.method);
      if (params.blacklisted) q.set("blacklisted", params.blacklisted);
      if (params.limit) q.set("limit", String(params.limit));
      if (params.offset) q.set("offset", String(params.offset));
      return apiRequest<MerchantMapAdminDTO[]>(API_BASE, `/admin/learned-merchants?${q.toString()}`);
    }
    
    export async function blacklistMerchant(merchantId: string, reason: string): Promise<void> {
      await apiRequest<void>(API_BASE, `/admin/learned-merchants/${merchantId}/blacklist`, {
        method: "POST", body: JSON.stringify({ reason }),
      });
    }
    
    export async function unblacklistMerchant(merchantId: string): Promise<void> {
      await apiRequest<void>(API_BASE, `/admin/learned-merchants/${merchantId}/unblacklist`, { method: "POST" });
    }
    
    export async function getModerationQueue(status_filter?: string): Promise<ModerationLogAdminDTO[]> {
      const q = status_filter ? `?status_filter=${status_filter}` : "";
      return apiRequest<ModerationLogAdminDTO[]>(API_BASE, `/admin/moderation${q}`);
    }
    
    export async function confirmModerationAction(logId: string): Promise<void> {
      await apiRequest<void>(API_BASE, `/admin/moderation/${logId}/confirm`, { method: "POST" });
    }
    
    export async function revertModerationAction(logId: string): Promise<void> {
      await apiRequest<void>(API_BASE, `/admin/moderation/${logId}/revert`, { method: "POST" });
    }
    ```
