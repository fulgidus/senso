    test('full user journey: auth → upload → profile → coach', async ({ page, account }) => {
      // 1. LOGIN
      await page.goto('/');
      await page.getByLabel(/email/i).fill(account.email);
      await page.getByLabel(/password/i).fill(account.password);
      await page.getByRole('button', { name: /accedi|login/i }).click();
      await expect(page).toHaveURL(/\/(chat|home|profilo)/);

      // 2. Navigate to file upload (Files tab or upload CTA)
      await page.getByRole('link', { name: /profilo|profile/i }).click();
      // Click upload / add document
      await page.getByRole('button', { name: /carica|upload/i }).click();
    });
