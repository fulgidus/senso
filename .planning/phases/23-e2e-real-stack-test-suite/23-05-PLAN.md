# Plan 23-05 - Mobile E2E: Full Journey on iPhone 14 Viewport

## Objective

Run the full happy-path journey on iPhone 14 Playwright emulation (webkit, 390×664,
touch events). Key assertions: input visible after keyboard open, cards render without
horizontal overflow, voice button has ≥ 44px tap target. Uses the `real-stack-mobile`
Playwright project configured in Plan 23-01.

<task id="23-05-01" title="Mobile E2E: auth + upload on iPhone 14">
  <objective>
    Run auth and file upload flow on iPhone 14 device emulation. Verify tap targets
    and layout adapt correctly to 390px width.
  </objective>
  <read_first>
    - senso/e2e/real-stack/fixtures.ts - account fixture, loginAs helper
    - senso/e2e/real-stack/user-journey.spec.ts - desktop journey to mirror
    - 23-RESEARCH.md - iPhone 14 exact viewport specs
  </read_first>
  <action>
    Create `senso/e2e/real-stack/mobile-journey.spec.ts`:

    ```ts
    import { test, expect } from './fixtures';

    // This file runs in the 'real-stack-mobile' project (iPhone 14, webkit)

    test('mobile: login + navigate to profile', async ({ page, account }) => {
      await page.goto('/');

      // Verify viewport is 390px
      const viewport = page.viewportSize();
      expect(viewport?.width).toBe(390);

      await page.getByLabel(/email/i).fill(account.email);
      await page.getByLabel(/password/i).fill(account.password);
      await page.getByRole('button', { name: /accedi|login/i }).tap();

      await expect(page).toHaveURL(/\/(chat|home|profilo)/);
    });

    test('mobile: nav items have ≥44px tap targets', async ({ page, account }) => {
      await loginAs(page, account);
      await page.goto('/');

      // Check all main nav items
      const navButtons = page.getByRole('navigation').getByRole('button');
      const count = await navButtons.count();
      for (let i = 0; i < count; i++) {
        const box = await navButtons.nth(i).boundingBox();
        if (box) {
          expect(box.height).toBeGreaterThanOrEqual(44);
          expect(box.width).toBeGreaterThanOrEqual(44);
        }
      }
    });

    test('mobile: upload file on 390px screen', async ({ page, account }) => {
      await loginAs(page, account);
      await page.goto('/profilo');

      // Upload button accessible on mobile
      const uploadBtn = page.getByRole('button', { name: /carica|upload/i });
      await expect(uploadBtn).toBeVisible();

      const box = await uploadBtn.boundingBox();
      expect(box?.height).toBeGreaterThanOrEqual(44); // tap target
    });
    ```
  </action>
  <acceptance_criteria>
    - Tests run in webkit with iPhone 14 viewport (390×664)
    - Login works with `.tap()` (touch events)
    - Nav buttons have ≥ 44px height
    - Upload button visible and accessible
  </acceptance_criteria>
</task>

<task id="23-05-02" title="Mobile E2E: chat input + cards + voice button assertions">
  <objective>
    Verify: chat input visible when keyboard simulated open, coach cards don't overflow
    390px, voice button has ≥ 44px tap target, profile tabs use grid layout.
  </objective>
  <read_first>
    - senso/e2e/real-stack/mobile-journey.spec.ts - from 23-05-01
    - 23-RESEARCH.md - keyboard simulation via setViewportSize
    - senso/src/features/coaching/ChatScreen.tsx - input selector
  </read_first>
  <action>
    Extend `mobile-journey.spec.ts`:

    ```ts
    test('mobile: chat input visible above simulated keyboard', async ({ page, account }) => {
      await loginAs(page, account);
      await page.goto('/chat');

      // Simulate keyboard open (shrink viewport by ~40%)
      await page.setViewportSize({ width: 390, height: 390 });

      // Chat input must still be in viewport (not behind keyboard)
      const input = page.getByRole('textbox', { name: /messaggio|message/i });
      await expect(input).toBeInViewport();

      // Restore
      await page.setViewportSize({ width: 390, height: 664 });
    });

    test('mobile: coach response cards do not overflow 390px', async ({ page, account }) => {
      await loginAs(page, account);
      await page.goto('/chat');

      // Send a message (LLM stub returns a response with cards)
      await page.getByRole('textbox').fill('Posso comprare qualcosa?');
      await page.getByRole('button', { name: /invia/i }).tap();

      await expect(page.getByTestId('assistant-bubble')).toBeVisible({ timeout: 15_000 });

      // Check no horizontal overflow at 390px
      const bodyWidth = await page.evaluate(() => document.body.scrollWidth);
      const viewportWidth = page.viewportSize()?.width ?? 390;
      expect(bodyWidth).toBeLessThanOrEqual(viewportWidth + 2); // 2px tolerance

      // Resource cards in horizontal scroll strip (not overflowing the layout)
      const cardStrip = page.getByTestId('resource-cards-strip');
      if (await cardStrip.count() > 0) {
        const box = await cardStrip.boundingBox();
        expect(box?.width).toBeLessThanOrEqual(viewportWidth);
      }
    });

    test('mobile: voice button ≥44px tap target', async ({ page, account }) => {
      await loginAs(page, account);
      await page.goto('/chat');

      const voiceBtn = page.getByRole('button', { name: /voce|parla|mic/i });
      if (await voiceBtn.count() > 0) {
        const box = await voiceBtn.boundingBox();
        expect(box?.width).toBeGreaterThanOrEqual(44);
        expect(box?.height).toBeGreaterThanOrEqual(44);
      }
    });

    test('mobile: profile tabs use grid layout at 390px', async ({ page, account }) => {
      await loginAs(page, account);
      await page.goto('/profilo');

      // Tab bar should be a grid on mobile (not a horizontal scroll strip)
      const tabGrid = page.locator('.grid.grid-cols-2');
      await expect(tabGrid).toBeVisible();

      // No horizontal overflow
      const bodyWidth = await page.evaluate(() => document.body.scrollWidth);
      expect(bodyWidth).toBeLessThanOrEqual(390 + 2);
    });
    ```
  </action>
  <acceptance_criteria>
    - Chat input in viewport when keyboard simulated (390×390)
    - `document.body.scrollWidth` ≤ 390px (no horizontal overflow on any page)
    - Resource cards horizontal strip bounded within viewport width
    - Voice button box ≥ 44×44px (if present)
    - Profile tab grid visible at mobile breakpoint
    - All mobile tests pass in `npx playwright test --project=real-stack-mobile`
  </acceptance_criteria>
</task>
