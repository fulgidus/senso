```python
"""
SENSO Platform hookspecs — the ONLY plugin interface surface.

Four hooks. No more. Adding hooks requires a platform version bump.

Domain plugins implement the hooks they need:
  - authenticate_request: REQUIRED for any domain with auth
  - process_extraction:   optional — enrich extracted document data
  - categorize_transactions: optional — add domain categories
  - enrich_response:      optional — post-process LLM coaching response

CORE-07: pluggy provides hook interfaces for extraction, categorization, enrichment, auth.
CORE-09: authenticate_request hookspec — domain plugin owns all auth logic.
"""

from __future__ import annotations

import pluggy

PROJECT_NAME = "senso"

hookspec = pluggy.HookspecMarker(PROJECT_NAME)
hookimpl = pluggy.HookimplMarker(PROJECT_NAME)


class SensoPlatformSpec:
    """Platform hookspec — implemented by domain plugins."""

    @hookspec(firstresult=True)
    def authenticate_request(self, token: str) -> str | None:
        """
        Validate a bearer token and return the platform identity_id, or None.

        CORE-09: Platform calls this on every authenticated request.
        Domain plugin (e.g. personal_finance) validates JWT and returns identity_id.
        Platform NEVER inspects token contents directly.

        firstresult=True: first non-None return wins (only one auth plugin per domain).
        """

    @hookspec
    def process_extraction(self, doc_type: str, raw_data: dict) -> dict:
        """
        Domain-specific post-processing of an extracted document.
        Return enriched/modified raw_data dict.
        """

    @hookspec
    def categorize_transactions(self, transactions: list[dict]) -> list[dict]:
        """
        Enrich a list of transaction dicts with domain-specific categories.
        Return the modified transaction list.
        """

    @hookspec(firstresult=True)
    def enrich_response(self, response: dict, context: dict) -> dict | None:
        """
        Domain-specific post-processing of a coaching LLM response dict.
        Return modified response or None to pass through unchanged.
        firstresult=True: first non-None return wins.
        """
```

```python
"""
FilterPipeline — executes ordered input/output filter chains from manifest config.

Built-in filter types:
  safety_scan      — SafetyScanner regex rules (hard_boundaries_ref)
  pii_scrub        — strip user PII tokens from output text
  domain_compliance — pass-through placeholder; domain plugins add logic via enrich_response hook

AI-01/AI-02: input/output filter pipelines.
"""

from __future__ import annotations

import logging
from dataclasses import dataclass, field
from typing import Any

from app.domain.manifest import FilterDef, FilterPipelineConfig

logger = logging.getLogger(__name__)


@dataclass
class FilterResult:
    passed: bool
    text: str
    blocked: bool = False
    reason: str | None = None
    severity: str | None = None


class FilterPipeline:
    """
    Executes an ordered list of FilterDefs against input or output text.

    Each filter can: transform text, flag a warning, or block entirely.
    First block wins — pipeline short-circuits.
    """

    def __init__(
        self,
        config: FilterPipelineConfig,
        scanner: Any | None = None,   # SafetyScanner | None
    ) -> None:
        self._input_filters = config.input
        self._output_filters = config.output
        self._scanner = scanner

    def run_input(self, text: str, context: dict[str, Any] | None = None) -> FilterResult:
        return self._run(text, self._input_filters, context or {})

    def run_output(self, text: str, context: dict[str, Any] | None = None) -> FilterResult:
        return self._run(text, self._output_filters, context or {})

    def _run(
        self,
        text: str,
        filters: list[FilterDef],
        context: dict[str, Any],
    ) -> FilterResult:
        current_text = text
        for f in filters:
            result = self._apply(f, current_text, context)
            if result.blocked:
                return result
            current_text = result.text
        return FilterResult(passed=True, text=current_text)

    def _apply(
        self, f: FilterDef, text: str, context: dict[str, Any]
    ) -> FilterResult:
        if f.type == "safety_scan":
            return self._safety_scan(text)
        elif f.type == "pii_scrub":
            return self._pii_scrub(text, context)
        elif f.type == "domain_compliance":
            # Placeholder: domain plugins add logic via enrich_response hook (Phase 39)
            return FilterResult(passed=True, text=text)
        else:
            logger.warning("FilterPipeline: unknown filter type '%s' — skipping", f.type)
            return FilterResult(passed=True, text=text)

    def _safety_scan(self, text: str) -> FilterResult:
        if not self._scanner:
            return FilterResult(passed=True, text=text)
        result = self._scanner.scan_input(text)
        if not result.safe:
            return FilterResult(
                passed=False,
                text=result.substitute_message or text,
                blocked=result.severity in ("block", "temp_ban", "hard_ban"),
                reason=result.reason,
                severity=result.severity,
            )
        return FilterResult(passed=True, text=result.text if hasattr(result, "text") else text)

    def _pii_scrub(self, text: str, context: dict[str, Any]) -> FilterResult:
        # Delegate to existing PII scrub logic from coaching/safety.py
        # context may contain user_profile and current_user for token extraction
        try:
            from app.coaching.safety import scrub_profile_pii  # noqa: PLC0415
            user_message = context.get("user_message", "")
            current_user = context.get("current_user", {})
            profile = context.get("profile_snapshot", {})
            scrubbed_response = scrub_profile_pii({"message": text}, user_message, current_user, profile)
            return FilterResult(passed=True, text=scrubbed_response.get("message", text))
        except Exception as exc:
            logger.warning("FilterPipeline: PII scrub failed: %s", exc)
            return FilterResult(passed=True, text=text)
```

```python
"""
BehaviorLayer — loads domain behavior directives from manifest refs.

Replaces hardcoded PERSONAS_DIR paths in CoachingService.
AI-03: ethos, boundaries, allowlist from Domain Manifest.
AI-07: SafetyScanner from hard_boundaries_ref.
"""

from __future__ import annotations

import logging
from pathlib import Path

from app.coaching.safety import SafetyScanner
from app.domain.manifest import BehaviorConfig, DomainManifest

logger = logging.getLogger(__name__)


class BehaviorLayer:
    """
    Loaded once at startup from manifest.behavior refs.
    Provides locale-aware behavior text and a domain-scoped SafetyScanner.
    """

    def __init__(self, manifest: DomainManifest, manifest_dir: Path) -> None:
        self._manifest_dir = manifest_dir
        behavior = manifest.behavior
        locale_default = manifest.domain.locale_default

        self._ethos = self._load_locale_files(behavior.ethos_ref, locale_default)
        self._boundaries = self._load_locale_files(behavior.boundaries_ref, locale_default)
        self._allowlist = self._load_locale_files(behavior.allowlist_ref, locale_default)

        if behavior.hard_boundaries_ref:
            boundaries_path = manifest_dir / behavior.hard_boundaries_ref
            self._scanner = SafetyScanner(boundaries_path=boundaries_path)
            logger.info("BehaviorLayer: SafetyScanner loaded from %s", boundaries_path)
        else:
            self._scanner = None
            logger.warning("BehaviorLayer: no hard_boundaries_ref — SafetyScanner disabled")

    def _load_locale_files(
        self, ref: str | None, locale_default: str
    ) -> dict[str, str]:
        """Load a behavior file for default locale + 'en' fallback."""
        if not ref:
            return {}
        result: dict[str, str] = {}
        base_path = self._manifest_dir / ref
        # Load default locale
        if base_path.exists():
            result[locale_default] = base_path.read_text(encoding="utf-8")
        # Load locale variants: ethos.en.md from ethos.md ref
        stem = base_path.stem
        suffix = base_path.suffix
        for locale in ("en", "it"):
            if locale == locale_default:
                continue
            locale_path = base_path.parent / f"{stem}.{locale}{suffix}"
            if locale_path.exists():
                result[locale] = locale_path.read_text(encoding="utf-8")
        return result

    def get_ethos(self, locale: str) -> str:
        return self._ethos.get(locale) or next(iter(self._ethos.values()), "")

    def get_boundaries(self, locale: str) -> str:
        return self._boundaries.get(locale) or next(iter(self._boundaries.values()), "")

    def get_allowlist(self, locale: str) -> str:
        return self._allowlist.get(locale) or next(iter(self._allowlist.values()), "")

    @property
    def scanner(self) -> SafetyScanner | None:
        return self._scanner
```

```python
"""PluginManager factory — builds the platform-wide pluggy PluginManager."""

from __future__ import annotations

import importlib
import logging

import pluggy

from app.platform.hookspecs import PROJECT_NAME, SensoPlatformSpec

logger = logging.getLogger(__name__)


def build_plugin_manager(plugin_ref: str | None = None) -> pluggy.PluginManager:
    """
    Build and return the platform PluginManager.

    plugin_ref: dotted Python path to the domain auth plugin class (from AuthConfig.plugin_ref).
                If None, manager is built with no plugins registered.
    """
    pm = pluggy.PluginManager(PROJECT_NAME)
    pm.add_hookspecs(SensoPlatformSpec)

    if plugin_ref:
        try:
            module_path, class_name = plugin_ref.rsplit(".", 1)
            module = importlib.import_module(module_path)
            plugin_class = getattr(module, class_name)
            pm.register(plugin_class())
            logger.info("PluginManager: registered plugin '%s'", plugin_ref)
        except Exception as exc:
            logger.warning(
                "PluginManager: failed to load plugin '%s': %s — continuing without plugin",
                plugin_ref, exc,
            )

    return pm
```

```python
@app.middleware("http")
async def domain_auth_middleware(request: Request, call_next):
    """
    CORE-09: Call domain authenticate_request hookspec on every request.
    Stores identity_id in request.state.identity_id (None if unauthenticated).
    Platform never inspects token contents.
    """
    # Skip auth for health checks, webhook (has own HMAC auth), docs
    if request.url.path in ("/health", "/docs", "/openapi.json", "/redoc"):
        request.state.identity_id = None
        return await call_next(request)

    token = None
    auth_header = request.headers.get("Authorization", "")
    if auth_header.startswith("Bearer "):
        token = auth_header[7:]

    identity_id: str | None = None
    if token and hasattr(request.app.state, "plugin_manager"):
        pm = request.app.state.plugin_manager
        try:
            identity_id = pm.hook.authenticate_request(token=token)
        except Exception:
            identity_id = None

    request.state.identity_id = identity_id
    return await call_next(request)
```
